དེ་མན་ཆད་ནི་སྐྱེ་བ་སྤྲུལ་སྐུའི་རྣལ་འབྱོར་ཏེ་མངོན་བྱང་ལྷག་མ་གསུམ་གྱི་མཐར་སྐད་ཅིག་མའི་སྔགས་བརྗོད་པས་སྣ་ཚོགས་པདྨའི་སྟེང་དུ་རྡོ་རྗེའི་ས་གཞི་དང་བསྲུངས་འཁོར་དུར་ཁྲོད་སོགས་རྟེན་དང་བརྟེན་པའི་དཀྱིལ་འཁོར་གོ་ཆ་དང་བཅས་པ་གཅིག་ཆར་གྲུབ་ནས་རིམ་གྱི་གསལ་གཏབ་སྟེ་ཡན་ལག་བཞི་རྫོགས་མཆོད་དོ། བཟླས་པའི་མཐར་ཉེར་བསྡུ་སྨོན་ལམ་དང་བཅས་པ་ནི་བདག་བསྐྱེད་ཡིན་ལ་ཐུན་མཚམས་སུ་མདུན་བསྐྱེད་བསྒྲུབ་མཆོད་བྱེད་པ་ནི། ཕྱི་དཀྱིལ་ཡོད་པའི་རྣམ་གཞག་ཡིན་པས་འདི་ལ་མི་རུང་བས་ན། མདུན་བསྐྱེད་སྤྲོ་ཞིང་དགོས་ན། ཉེར་བསྡུའི་གོང་དུ་རང་གི་སྙིང་ཁའི་གཙོ་བོ་ལས་གཙོ་བོ་གཉིས་པ་ལྷན་སྐྱེས་ཡབ་ཡུམ་དུ་བྱེ་ནས་དཀྱིལ་འཁོར་གྱི་མདུན་དུ་གནས་པས་མཆོད་པ་རྒྱས་པ་དང་བདག་འཇུག་དང༌། དེར་སློབ་མ་འཇུག་པ་སོགས་བྱས་པའི་མཐར་ཉེར་བསྡུ་བྱ་བ་ཡིན་ནོ། ། དེ་ནས་ཐུན་མཚམས་སུ་ནི་གཏོར་མ་དང་ཁ་ཟས་སོགས་ཀྱི་སྤྱོད་ལམ་རྣམས་བྱའོ། ། གཉིས་པ་ནི། རི་བོ་ཚང་ཚིང་སྨན་ལྗོངས་དང༌། ། གདོད་ནས་གྲུབ་པའི་དུར་ཁྲོད་སོགས། །

རྣམ་སྣང་འཁོར་བསྒྱུར་རྡོར་སེམས་བརྩོན་འགྲུས་ཆེ། ། ཞི་བའི་ལས་མཛད་འོག་སྤྱོད་སྐུ་ཡིས་ཚོགས། ཁ་དང་འུག་གདོང་ཁྱི་དང་ཕག་གདོང་མ། ། བརྟན་མ་ཕོ་ཉ་མཆེ་ བ་འཇོམས་མ་སྟེ། ། ཕྱོགས་བཅུའི་གསང་རྣམས་སུ་དགའ་བ་དང་མཆོག་དགའ་དང་ཁྱད་པར་གྱི་དགའ་བ་དང༌། ལྷན་ཅིག་སྐྱེས་དགའི་ཡེ་ཤེས་སྐྱེས་པའི་བདེ་བ་དེས་སྟོང་པ་ངེས་པའི་བདེ་སྟོང་གི་ཡེ་ཤེས་དབང་གསུམ་པའི་ངོ་བོ་ཉམས་སུ་མྱོང་བར་གྱུར། རིག་མ་དེ་དགྱེས་ནས། དེང་ནས་བཟུང་ནས་བུ་ཁྱོད་ཀྱིས། ། ཇི་སྲིད་བྱང་ཆུབ་སྙིང་པོའི་བར། ། རྨིས་ལམ་བཞིན་དུ་སེམས་ཅན་གྱི། །
དོན་ནི་བདག་ཉིད་ཀུན་གྱིས་བྱོས། ། ཞེས་གསུངས་པར་གྱུར། དེ་ལྟར་ན་བྷ་གའི་དཀྱིལ་འཁོར་དུ་ཤེས་རབ་ཡེ་ཤེས་ཀྱི་དབང་ཐོབ། ཡིད་ཀྱི་དྲི་མ་སྦྱངས། ནོར་བུ་འགེངས་པའི་རིམ་པ་སོགས་ཕོ་ཉའི་ལམ་བསྒོམ་པ་ལ་དབང༌། ཐུགས་རྡོ་རྗེ་ཆོས་སྐུའི་ནུས་པ་མངོན་དུ་བྱས་སོ། ། དབང་བཞི་པ་ལ། མཎྜལ་འཕུལ། བདག་ཉིད་ཆེ་ཁྱོད་བཀའ་དྲིན་གྱིས། ། གཙོ་བོར་དབང་བསྐུར་རྣམ་གསུམ་ཐོབ། །
དེ་རིང་རིན་ཆེན་བཞི་པ་ཡང༌༑ ༑ བདག་ལ་བཀའ་དྲིན་སྐྱབས་ཏུ་གསོལ། ། ལན་གསུམ་གྱིས་གསོལ་བ་བཏབ༑ སློབ་དཔོན་གྱིས། འདི་ནི་རང་ཉིད་མཛེས་མ་སྟེ། ། ཕྱིར་རོལ་ཡུལ་ནི་སྤངས་པའོ། ། རིག་འདི་བོར་ན་དེ་ལས་ནི། ། གཞན་པའི་ཐབས་ཀྱིས་འཚང་མི་རྒྱ༑ ༑ བརྟུལ་ཞུགས་རྣམས་ཀྱི་བླ་མའི་བླ། ། རིགས་མའི་བརྟུལ་ཞུགས་སྤྱོད་ཅིག་བུ། །
ཞེས་སྤྱིར་རིགས་མ་དང༌། དགོས་ཕྱི་རོལ་བེམ་ཆོས་སྤངས་པའི་རང་སེམས་མཛེས་མ་སྟོང་གཟུགས་ཕྱག་རྒྱ་ཆེན་མོར་ཤར་བ་ལ་བདེ་སྟོང་སྦྱར་བའི་གཞི་དུས་ཀྱི་བདེ་མཆོག་ལམ་གྱི་ཇེ་བཟང་དུ་སོང་བའི་མཐར་ཐུག་གི་ཟུང་འཇུག་ནི་ཐོབ་བྱ་མཐར་ཐུག་དོན་གྱི་དབང་བཞི་པ་ཞེས་སམ། ཇོ་བོ་ཆེན་པོས་ལོ་ཧི་པའི་འགྲེལ་པར་གསུངས་པ་སྒྲུབ་ཐབས་སུ་བྲིས་པ་བརྗོད་ནས་ངོ་སྤྲོད། དེ་ལྟར་དོན་དམ་བྱང་ཆུབ་སེམས་ཀྱི་དཀྱིལ་འཁོར་དུ་དབང་བཞི་པ་ཚིག་དབང་རིན་པོ་ཆེ་ཐོབ༑ སྒོ་གསུམ་གྱི་དྲི་མའི་བག་ཆགས་སྦྱངས། བསམ་གྱིས་མི་ཁྱབ་པ་བསྒོམ་པ་ལ་དབང༌། ངོ་བོ་ཉི་སྐུའམ་ལྷན་ཅིག་སྐྱེས་པའི་སྐུའི་ནུས་པ་མངོན་བཞིན་དུ་བྱས་སོ༑ ༑ དེ་ནས། གཙོ་བོས་ཇི་ལྟར་བཀའ་སྩལ་པ། ། དེ་དག་ཐམས་ཅད་བདག་གིས་བགྱི། །
