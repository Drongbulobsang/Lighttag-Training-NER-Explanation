མ་ཡིན་ཀྱང་དུ་བ་ནི་མེ་ལས་བྱུང་ཞིང༌། དེར་འགག་པས་དུ་བ་རྟག་ཏུ་མེ་དང་འདྲེས་པས་སོ། ། དེ་བཞིན་དུ་དཀར་པོ་ལ་སོགས་པའི་རྟོག་པ་ཡང་སྐྱེ་མེད་ལས་སྔོན་རོལ་དུ་གྲུབ་པ་ཡང་མ་ཡིན་ལ་མི་འགག་པ་གནས་པ་ཡང་མ་ཡིན་ཏེ། འོན་ཀྱང་སྐྱེ་མེད་ལས་བྱུང་ཞིང་དེར་ཐིམ་པས་སྐྱེ་མེད་དང་རྒྱུན་དུ་འབྲེལ་བ་ཡིན་ནོ། ། དེ་ནས་དཀར་པོ་ལ་སོགས་པའི་རྟོག་པ་ཡང་དེ་ནས་དཀར་པོའི་བློ་འགགས་པའི་དུས་སུ་འགགས་པ་སྐྱེ་མེད་ནམ་མཁའ་བཞིན་ཞེས་བྱ་བ་ཡིན་ཏེ། ཐིམ་པས་ཐིམ་པ་བཙལ་བ་མི་རྙེད་པ་བཞིན་ནོ། ། དེ་ལ་ནམ་མཁའ་བཞིན་ཞེས་བྱ་བ་ནི། ནམ་མཁའ་ལ་ཐོག་མཐའ་དང་སྐྱེ་འགགས་མེད་པ་བཞིན་དུ་སེམས་ལ་ཡང་ཐོག་མཐའ་དང་སྐྱེ་འགག་མེད་དེ། མ་སྐྱེས་པའི་དུས་ནི་སྐྱེ་བ་ཡིན་ལ། དཀར་པོའི་བློ་སྐྱེས་པའི་དུས་སུ་ཡང༌།
འགྲོ་བ་དང་ཐིམ་པ་མི་འདུག་པས་སྐྱེ་བ་མེད་པ་ཉིད་ཡིན་པས་ན་ཆོས་ཐམས་ཅད་ནི་སྐྱེ་འགག་དང་བྲལ་བ་ནམ་མཁའ་ལྟ་བུ་ཡིན་ནོ། ། ཞེས་བློ་ཐག་བཅད་ལ། ཟུང་དུ་འཇུག་པའི་ངང་ལ་བློ་ཇི་ཙམ་གནས་ཚད་གཅིག་བཞག་གོ ། ཞེས་བརྗོད་ལ་དལ་གཅིག་བློ་གནས་སུ་གཞུག དེ་ནས་ཡང་རོལ་མོ་གུ་གུལ་དུད་པ་ཨོཾ་གསུམ་མ་བརྗོད། གཙོ་བོས་ཇི་ལྟར་དང༌། མཎྜལ། དེང་ནས་སོགས་བྱ། སློབ་མ་རྣམས་རང་གནས་སུ་བཏང་ལ། མདུན་བསྐྱེད་མཆོད་བསྟོད་བསྔོ་བ་སོགས་བྱ་ཞིང་རྗེས་བསྡུའོ། །
ཞེས་རྫོགས་རིམ་རྩ་དབུ་མའི་ཁྲིད་ཀྱི་ཆོས་ཉིད་དངོས་སུ་སྟོན་པའི་ཕྱག་ལེན་གྱི་རིམ་པ་གསལ་བར་བཤད་པ་གོ་བདེ་བཀླགས་ཆོག་མ་འདི་ཡང་ས་སྐྱ་པ་སྔགས་འཆང་ངག་དབང་ཀུན་དགའ་བསོད་ནམས་ཀྱིས་དཔལ་ས་སྐྱའི་གཞི་ཐོག་བླ་བྲང་དུ་སྦྱར་བའི་ཡི་གེ་པ་ནི་ཨོ་རྒྱན་སྐྱངས་ཀྱི་བགྱིས་པ་འདིས་ས་སྐྱ་པའི་མན་ངག་ཟབ་མོ་རྣམས་ཀྱི་བཀའ་མི་ཆད་པའི་རྒྱུར་གྱུར་ཅིག། ༈ །འདིའི་ཕྱག་ལེན་གྱི་བརྒྱུད་པ་ནི། ཆོས་རྗེ་ས་པཎྜི་ཏ་ཡན་གཞན་དང་འདྲ་བ་ལས། དེ་ནས་ཚོགས་སྒོམ་པ། ཉན་ཆེན་པ། བྲག་ཕུག་པ། ཆོས་རྗེ་བླ་མ་དམ་པ། ཡང་ན་ས་པཎ། འཕགས་པ། ཞང་དཀོན་མཆོག་དཔལ།
བྲག་ཕུག་པ། བླ་མ་དམ་པ། ཡང་ན་འཕགས་པ། པཎྜི་ཏ་རྡོ་རྗེ་ཤེས་རབ། འཕགས་རྒྱལ། བླ་མ་བཟང་པོ་བ། བླ་མ་དམ་པ། ཡང་ན་འཕགས་པ། ཞབ་མདོ་སྡེ་དཔལ། འཇམ་དབྱངས་རིན་རྒྱལ།
དེ་གཉིས་ལ་འཇམ་སྐྱ། རྗེ་དཔལ་ལྡན། བླ་མ་དམ་པ། ཡང་ན་ས་པཎ། ལར་སྒོམ་སེའོ། ། དོ་པ་སྟོན་ཚུལ་གསུམ་ལ། རྒྱ་ཚ་ལུང་མོང་པོས་གསན། མཁས་པ་ཡོན་ཏན་ཁྲི། དྲིན་ཆེན་སེངྒེ་བཟང་པོ། ཡར་ཀླུང་པ་སེངྒེ་རྒྱལ་མཚན།
མཁན་ཆེན་རྒྱལ་མཆོག་པ། འཇམ་དབྱངས་ནམ་མཁའ་རྒྱལ་མཚན། བདག་ཆེན་བློ་གྲོས་རྒྱལ་མཚན། རྗེ་རྡོ་རིང་པ། རྗེ་ཚར་ཆེན། རྗེ་མཁྱེན་བརྩེའི་དབང་ཕྱུག རྗེ་བྱམས་པ་རབ་བརྟན། རྗེ་སྙིང་རྗེ་བཟང་པོ། སྤྱན་སྔ་ཆོས་ཀྱི་སྤྱན་ལྡན། ཀུན་དགའ་དོན་གྲུབ།
དེས་བདག་ས་སྐྱ་པ་ཨཱ་ནནྡ་ལའོ། ། ཡང་ན་ས་པཎ། ཚོགས་སྒོམ་པ། ཉན་ཆེན། དཀར་པོ་བྲག་པ་རིན་ཆེན་སེངྒེ། དེ་ལའང་ཡར་ཀླུང་པ་སེངྒེ་རྒྱལ་མཚན་གྱིས་གསན་པར་བཞེད་དེ། ཆོས་རྗེ་བླ་མ་དམ་པའི་ཞལ་ནས་ཆོས་རྗེ་བྲག་ཕུག་པའི་མན་ངག་ཟབ་མོ་རྩིས་ཆེ་ཤོས་གཅིག་ཡིན། ཕྱག་ལེན་མ་ནོར་བ་ཡིན་ཞེས་ནན་ཏན་དུ་གསུངས་སོ། ། ཞེས་པ་ཡར་ཀླུང་པས་འདིའི་ཡིག་ཆ་ལས་འབྱུང་ངོ༌།། བཀྲ་ཤིས།། །།
