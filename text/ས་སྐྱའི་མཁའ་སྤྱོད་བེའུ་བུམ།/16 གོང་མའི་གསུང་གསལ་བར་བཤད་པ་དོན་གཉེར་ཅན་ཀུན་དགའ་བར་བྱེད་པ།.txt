ཐོག་མར་རྒྱལ་བ་རྒྱ་མཚོ་མ་ལ་སོགས་པའི་བརྒྱུད་པའི་གསོལ་འདེབས་དང༌། རང་ལ་སྐུ་དྲིན་ཆེ་བའི་རྩ་བའི་བླ་མའི་སྐུ་བསྟོད་ཅི་ནུས་དང༌། དངོས་གཞི་སྐྱབས་འགྲོ་ནས་བཟུང༌། ཐུན་བསྡུ་བའི་བར་ཚར་ཆེན་ཆོས་ཀྱི་རྒྱལ་པོའི་ལུགས་འདི་ལ་ཚད་མར་བརྒྱུད་པས་དེའི་སྒྲུབ་ཐབས་ལྟར་ཉམས་སུ་བླངས། སྔོན་འགྲོ་སྐྱབས་འགྲོ་རྡོར་སེམས་བླ་མའི་རྣལ་འབྱོར་གསུམ་ཞག་གྲངས་རང་རང་གི་བློ་དང་བསྟུན་པ་བྱ། ཡང་དངོས་གཞི་ལམ་དུ་འགྲོ་མི་འགྲོ། སྔོན་འགྲོ་གནད་དུ་བསྣུན་མི་བསྣུན་ལ་རག་ལས་གསུངས་པས་ལས་སླར་མ་སོང་བ་བྱའོ། ། དེ་ནས་བདག་ལྷར་བསྐྱེད་པ་ནས་ལུས་དཀྱིལ་གསལ་འདེབས་ཡོ་ག་ཤུདྡྷོ྅འི་བར་ཤར་སྒོམ་དུ་ཉམས་སུ་བླངས། དེ་ནས་སྤྱིར་བཟླས་དམིགས་ནི། དཔུང་བཟང་གི་རྒྱུད་ལས།

དེ་དག་ལས་འོད་ཟེར་འཕྲོས་ལུས་ཀྱི་ནང་ཐམས་ཅད་མེར་གྱིས་གང་བས་ཚེ་རབས་འཁོར་བ་ཐོག་མ་མེད་པ་ནས་བསགས་པའི་ནད་གདོན་སྡིག་སྒྲིབ་ངན་སོང་དུ་སྐྱེ་འགྱུར་གྱི་ལས་ངན་བག་ཆགས་དང་བཅས་པ་ཐམས་ཅད་མཁའ་སྤྱོད་བསྒྲུབ་པའི་བར་ཆད་ཐམས་ཅད་ཤྭ་སྦུད་ཆེན་པོའི་སྣེ་ལ་སྦུར་མི་ཆགས་པའམ། ཉི་མ་ཤར་བས་མུན་པ་རང་སངས་སོང་བ་བཞིན་དུ་དེ་དག་སངས་ཀྱིས་དག ལུས་འོད་ཀྱི་རང་བཞིན་དུ་བྱས། ཡང་སྙིང་གའི་བྃ་ཡིག་སྔགས་ཕྲེང་དང་བཅས་པ་ལས་འོད་ཟེར་ལུས་ཀྱི་ཆ་ཤས་ཐམས་ཅད་ནས་ཕྱིར་འཕྲོས་སེམས་ཅན་ཐམས་ཅད་ལ་ཕོག་པས་དེ་དག་གི་ནད་གདོན་སོགས་སྔར་བཞིན་སྦྱངས། དེ་ཐམས་ཅད་རྗེ་བཙུན་མའི་གོ་འཕང་ལ་བཀོད་འོད་དུ་ཞུ། ཚུར་འདུས་རང་གི་སྙིང་གའི་བྃ་ཡིག་སྔགས་ཕྲེང་དང་བཅས་པ་དང་ལུས་ཀྱི་ཆ་ཤས་ཐམས་ཅད་ནས་སིབ་སིབ་ཐིམ་པས་རང་ཉིད་རྗེ་བཙུན་མ་སྔར་བས་ཀྱང་ཇེ་གསལ་བྱིན་རླབས་ཇེ་ཆེར་གྱུར་བར་མོས་ནས། རང་གི་སྔགས་བཟླ་བ་བཞིན། དགྲ་གཉེན་བར་མའི་སེམས་ཅན་ལུས་ཀྱི་སྲིན་བུ་ཚུན་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་སྐུར་གྱུར་གསལ་བས་སྔགས་ཀྱི་སྒྲ་དི་རི་རི་འདྲེན་པར་བསམས་ལ། ཨོཾ་གསུམ་མ་བཟླས་པའི་སྐྱོན་བརྒྱད་སོགས་དང་བྲལ་བར་བཟླས་སམ། གཙོ་ཆེར་དམིགས་པ་དེ་ཉིད་སྦྱང་ཡང་དེ་སྔར་ལྟར་འོད་ཟེར་འཕྲོས་པའི་སྣེ་ལ་མཆོད་ཡོན་སོགས་ཉེར་མཆོད་ཕྱི་ནང་གི་མཆོད་པའི་ལྷ་མོ་རྒྱལ་སྲིད་སྣ་བདུན།
འདོད་ཡོན་ལྔ་སོགས་སྤྲོས་ཏེ་ཕྱོགས་བཅུའི་བླ་མ་ཡི་དམ་སངས་རྒྱས་བྱང་སེམས་ལ་སོགས་མཆོད་པས་མཆོད། བསྟོད་པས་མཉེས་པར་བྱས། དེ་དག་གི་ཐུགས་རྗེ་བྱིན་རླབས་ཐམས་ཅད་འོད་ཀྱི་རྣམ་པར་སྤྱན་དྲངས་ལུས་ཀྱི་ཆ་ཤས་ཐམས་ཅད་ནས་ཐིམ་པས་དེ་ཐམས་ཅད་ཀྱི་བྱིན་རླབས་མ་ལུས་པ་རང་རྒྱུད་ལ་ཞུགས་པས་དེ་དག་གི་སྐུ་གསུང་ཐུགས་དང་རང་གི་ལུས་ངག་ཡིད་གསུམ་དབྱེར་མེད་དུ་གྱུར་སོང་སྙམ་པའི་ངང་ནས་འཛིན་པ་མ་ཞུགས་པར་སྔགས་བཟླ། དེས་ཀྱང་རྣམ་རྟོག་འཕྲོ་བ་དང་སེམས་དུབ་པ་སོགས་བྱུང་ན་རང་ཉིད་རྣལ་འབྱོར་མར་གསལ་ལེགས་པར་བཏབ། ལྷ་སྐུ་སྣང་ལ་རང་བཞིན་མེད་པ་དེ་ཉིད་ལ་དམིགས་པ་གཏད་དེ་འཛིན་མེད་དུ་བཟླ། རེས་འགའ་སྔགས་ཀྱི་སྒྲ་གྲགས་སྟོང་སྐྱེ་བ་མེད་པ་ལ་སེམས་གཏད། སྒྲ་དེ་གང་ནས་འོང༌། གང་དུ་འགྲོ། གང་ན་གནས། སེམས་ཀྱིས་བརྟགས་ཏེ་དེའི་ངང་ལ་བཟླ།

ཏིལ་ཉུངས་གང་རིགས་ལ་ཨོཾ་གསུམ་མའི་གྲངས་ཅི་འགྲོ་བསགས་པས་ཡོང་ཞེས་བླ་མ་དམ་པ་རྣམས་ཀྱི་གསུང་ལས་ཐོས་པ་ལགས། ལྷ་སྔགས་དངོས་གཞིའི་བསྲེག་རྫས་འབུལ་བའི་དམིགས་པ་ནི། རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མར་ཕུལ་བར་བསམ་ཞིང༌། དེ་ནས་ཡམ་ཤིང་མེ་འབར་བའི་དཀྱིལ་དུ། མར་དང་ཤ་ལྔ་བདུད་རྩི་ལྔ་ཟས་མཆོག་རྣམས་ཞལ་དུ། ཀུ་ཤ་དང་དཱུརྺའི་མྱུ་གུ་དབུའི་གཙུག་ཏོར་ལ། འབྲས་སོགས་གོང་བུའི་རྫས་རྣམས་ཕྱག་ཏུ་ཕུལ་བར་བསམ། དེ་ཡང་ནམ་མཁའི་ཁམས་ཐམས་ཅད་རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མས་དབུ་བྱས་པའི་སངས་རྒྱས་བྱང་སེམས་ཀྱི་ཚོགས་ཀྱིས་གང་བ་སྐུ་མདོག་དཀར་པོ་ཕྱག་ན་ཡེ་ཤེས་ཀྱི་བདུད་རྩིས་གང་བའི་བུམ་པ་འཛིན་པས་བདག་ལ་དབང་བསྐུར་ཞིང་སྡིག་སྒྲིབ་མ་ལུས་པ་དག་ནས། ལུས་ཡེ་ཤེས་ཀྱི་བདུད་རྩིས་མེར་གྱིས་ཁེངས་ཏེ་རྗེ་བཙུན་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་གོ་འཕང་མངོན་དུ་བྱས་བྱུང་སྙམ་པའི་ཏིང་ངེ་འཛིན་མ་ཡེངས་པར་བྱས་ཤིང༌། མདུན་བསྐྱེད་ཀྱི་སྐུ་ལས་མདངས་དང་འོད་ཟེར་དཀར་པོ་འཕྲོས་པས་བདག་དང་བསྒྲུབ་བྱ་ལ་ཕོག་པས་སྡིག་སྒྲིབ་ཐམས་ཅད་སྦྱངས་པར་བསམ།
