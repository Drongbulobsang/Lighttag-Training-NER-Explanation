སྤྱན་རས་གཟིགས་ཀྱིས་ལན་སྟོན་པ་དང་ཤཱ་རིའི་བུས་དྲི་བ་བྱེད་པར་བྱིན་གྱིས་བརླབས་པས་སོ། ། ཤཱ་རིའི་བུས་དྲི་པ་ནི། ཤཱ་རིའི་བུས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོའི་སྤྱོད་པ་སྤྱད་པར་འདོད་པ་དེས་ཇི་ལྟར་བསླབ་པར་བྱ་ཞེས་དྲིས་སོ། ། དྲི་བ་འདི་ལ་རིགས་ཀྱི་བུ་ཞེས་བྱ་བ་ཡོད་ཀྱི་རིགས་ཀྱི་བུ་མོ་ཞེས་བྱ་བ་མེད་ཅེས་སྨྲ་བ་ནི་ནོར་བ་སྟེ། རྒྱ་དཔེ་ལ་ཡོད་པས་སོ། ། དེ་ནས་སྤྱན་རས་གཟིགས་ཀྱིས་དེའི་ལན་ཤེར་ཕྱིན་ལ་བསླབ་ཚུལ་བསྟན་པ་ནི། དེའི་ཚེ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་གིས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་ལ་འདི་སྐད་ཅེས་སྨྲས་སོ། ། ཤཱ་རིའི་བུ་རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་ལ་ལ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོའི་སྤྱོད་པ་སྤྱད་པར་འདོད་པ་དེས་འོག་ནས་འབྱུང་བ་འདི་ལྟར་རྣམ་པར་བལྟ་བར་བྱ་སྟེ་ཞེས་མཚམས་སྦྱར་ནས། ལྟ་ཚུལ་དངོས་བསྟན་པ་ནི།
གཟུགས་སྟོང་པའོ་ཞེས་སོགས་གསུངས་ཏེ། ཟབ་མོ་ཁྱད་པར་བཞི་ལྡན་བསྟན་ཏོ། ། འདི་ནི་ཆུ་ཟླའི་དཔེ་ཤེས་པར་བྱ་སྟེ། དཔེར་ན་ཆུ་ནང་དུ་ཟླ་བའི་གཟུགས་བརྙན་སྣང་བ་ན། ཆུ་ནང་དུ་ཟླ་བ་སྣང་བ་ཙམ་མ་འགག་པར་ཡོད་པ་དང༌། ཟླ་བ་སྣང་བའི་དུས་ནས་ཆུ་ནང་དུ་ཟླ་བ་ཡོད་མ་མྱོང་བ་དང་ཟླ་བར་སྣང་བ་དང༌། ཟླ་བའི་རང་གི་ངོ་བོར་ཡོད་མ་མྱོང་བ་ཐ་མི་དད་པ་དང༌། ཟླ་བའི་མཚན་ཉིད་ཤེས་ནས་རང་སར་གྲོལ་བའོ། ། དཔེ་དེ་བཞིན་དུ་གཟུགས་སུ་སྣང་བའི་དུས་ཉིད་ནས་གཟུགས་ཀྱི་རང་གི་ངོ་བོས་སྟོང་པ་སྟེ། འདི་ནི་གཟུགས་སྟོང་པའོ་ཞེས་པའི་དོན་ནོ། །
