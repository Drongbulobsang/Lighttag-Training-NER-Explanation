བློ་སྦྱོང་ཚིགས་རྐང་བརྒྱད་མ་ལོ་རྒྱུས་དང་བཅས་པ་བཞུགས་སོ། ། བླ་མ་དམ་པ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། ། དགེ་བཤེས་འཆད་ཀ་བའི་ཞལ་ནས། བདག་བཀའ་གདམས་པ་རྣམས་ལ་དད་པ་འདི་དང་པོ་སློབ་དཔོན་ལྕགས་ཞིང་བ་ལ་ཚིག་རྐང་བརྒྱད་པ་འདི་ཐོས་ནས། དེ་ནས་རླུང་ཤོད་དགེ་གོང་བར་དུ་འདིའི་བཙང་དར་བཅད་ནས་ཚིག་རྣམས་ལ་སྔོ་བགྲང་བག་རེ་བྱས་པ་ལས་རྒྱུད་ལ་སྐྱེས་ནི་མི་བདོག སྐྱེས་ན་འདི་མིན་གཅིག་འོང་བ་ལ་གསུངས། དང་པོ་ནི། འགྲོ་འོང་བྱེད་ཙ་ན་དགྲ་བྱུང་གིས་དོགས་པ་ལ་འདི་དྲན་པས་ཕན་པར་བྱུང་། བློ་རྒོད་དུ་ཐལ་བས་མི་ལ་གནས་ཁང་གཉའ་བ་ཤ་སྟག་དགོས་པར་སོང་བས། དེ་ཙ་ན་གནས་གཤའ་མ་མ་རྙེད་པ་དང་།

ཡང་འདི་སྐད་དུ། བདག་ནི་ལས་ངན་གྱི་རྩ་བ་སྟེ། ། རྒྱངས་ཏེ་བསྐྱུར་བའི་ཆོས་གཅིག་ཡིན། ། གཞན་ནི་བྱང་ཆུབ་ཀྱི་འབྱུང་ཁུངས་ཏེ། ། བཅངས་ཀྱི་ལེན་པའི་ཆོས་གཅིག་ཡིན། ། སྤང་གཉེན་མཇུག་སྡུད་ཀྱི་ཆོས་གཉིས་ཏེ། ། མཐའ་འཁོབ་འདུལ་ན་དགོས་པ་ཡིན། ། འཁོར་ངན་ལོག་སྒྲུབ་བཟོད་པ་ལ། ། སྙིགས་མའི་དུས་སུ་མཁོ་བ་ཡིན། ། ཞེས་པའི་དོན་ནི།

རྒྱངས་ཀྱིས་བསྐྱུར་ལ་ལྷོད་ཀྱིས་གློད། ། ཤིག་གིས་ཤིག་ལ་འབོལ་ལེ་བཞག ། སྟོང་ཉིད་རྒྱས་འདེབས་བཞི་ཡིན་ཏེ། ། མཐའ་འཁོབ་འདུལ་ན་དགོས་པ་ཡིན། ། འཁོར་ངན་ལོག་སྒྲུབ་བཟོད་པ་ལ། ། སྙིགས་མའི་དུས་སུ་མཁོ་བ་ཡིན། ། དེ་ལྟར་ཉམས་སུ་བླངས་གྱུར་ན། ། སྐྱེས་བུ་མི་འཆིང་གྲོལ་བར་འགྱུར། ། ཅེས་པའི་དོན་ནི་གཟུང་འཛིན་གྱི་རྟོག་པ་ཐམས་ཅད་ཕར་རྒྱངས་ཏེ་བསྐྱུར་ལ། སྤྲོས་བྲལ་ཆོས་ཀྱི་དབྱིངས་སུ་ཤེས་པ་གསལ་ལ་རྟོག་པ་མེད་པར་ལྷོད་དེ་གློད་ལ་གང་གིས་ཀྱང་མ་བཅིངས་པར་བཞག་པར་བྱའོ། །
