ཁུ་ཇོ་སྲས་ནེ་ཙོ། རྔོག་ཇོ་སྲས་རྭ་མོ། གཉོས་ཇོ་སྲས་འོད་མ་དང་བཞིར་གྲགས་པའི་ནང་ནས་མཆོག་ཏུ་གྱུར་པ་འཁོན་ཇོ་སྲས་རྩེ་མོ་ཞེས་ཡོངས་སུ་གྲགས་སོ། ། དགུང་ལོ་ཉི་ཤུ་རྩ་དྲུག་བཞེས་པ་མེ་མོ་ཕག་གི་ལོ་ལ་མངའ་རིས་གུང་ཐང་ན་ལ་རྩེ་གནས་གསར་གྱི་གཙུག་ལག་ཁང་དུ་ཆོས་ལ་འཇུག་པའི་སྒོ་ཞེས་བྱ་བའི་བསྟན་བཅོས་ཆེན་པོ་ཐུགས་རྩོམ་གནང་། དེའི་ནང་དུ་བསྟན་པའི་རྩིས་ཀྱི་རྣམ་པར་གཞག་པ་སྐྱོན་ཐམས་ཅད་དང་བྲལ་བ་ཡང་མཛད། དེ་ལྟར་མཁས་པ་ཆེན་པོར་གྱུར་ནས་ནང་དུ་དེ་ཁོ་ན་ཉིད་ཀྱི་དོན་ལ་མཉམ་པར་འཇོག་ཅིང་། རྗེས་ཐོབ་ཏུ་ཡང་འཆད་པ་དང་། རྩོད་པ་དང་། རྩོམ་པའི་བྱ་བས་དུས་འདའ་ཞིང་། ཁྱད་པར་ཉིན་མཚན་ཀུན་ཏུ་ཡི་དམ་གྱི་ལྷ་ཀྱཻ་རྡོ་རྗེ་དང་།
