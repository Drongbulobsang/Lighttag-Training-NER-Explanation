བླ་མ་ལ་སངས་རྒྱས་དངོས་ཀྱི་འདུ་ཤེས་ཀྱིས་སྔར་ལྟར་མཆོད་ལ། བརྗོད་པར་འགྱུར་པའི་དོན་སེམས་ཤིང་། ཕྱོགས་བཅུ་ན་བཞུགས་པའི་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་དང་། སློབ་དཔོན་བདག་ལ་དགོངས་སུ་གསོལ། བདག་མིང་འདི་ཞེས་བགྱི་བས་ཚེ་འཁོར་བ་ཐོག་མཐའ་མེད་པ་ནས་ཐ་མ་ད་ལྟ་ལ་ཐུག་གི་བར་དུ་སྦྱིན་པའི་རང་བཞིན་ཅན་དང་། ཚུལ་ཁྲིམས་ཀྱི་རང་བཞིན་ཅན་དང་། སྒོམ་པའི་རང་བཞིན་ཅན་གྱི་དགེ་བའི་རྩ་བ་དེས། ཇི་ལྟར་སྔོན་གྱི་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་དང་། ས་ཆེན་པོ་ལ་རབ་དུ་བཞུགས་པའི་བྱང་ཆུབ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱིས་ཇི་ལྟར་ཐུགས་བསྐྱེད་པ་བཞིན་དུ། བདག་མིང་འདི་ཞེས་བགྱི་བས་ཀྱང་དུས་འདི་ནས་བཟུང་ནས་སྙིང་པོ་བྱང་ཆུབ་ལ་མཆིས་ཀྱི་བར་དུ།

དེ་ལས་འོད་ཟོར་སྔོན་པོ་གསལ་སིང་བ་བྱུང་། སྙིང་ཁའི་རྩ་མདུད་རྣམས་གྲོལ། ལྟེ་བའི་ཨ་གསོས། དེ་ལས་འོད་ཟེར་དམར་ལྗང་གསལ་སིང་བ་བྱུང་། ལྟེ་བ་ནས་རྐང་མཐིལ་ཡར་ཆད་ཀྱི་རྩ་མདུད་རྣམས་གྲོལ། བདེ་སིང་བ་ལ་སེམས་གཏད་པའི་ངང་ནས་ལྷན་སྐྱེས་སུ་ལམ་གྱི་བསྒོམ་མོ། ། གསུམ་པ་ལུས་དག་པ་ལྷ་སྐུར་བསྒོམ་པ་ནི་ལྷན་སྐྱེས་སུ་གསལ་བ་དེའི་ནོར་བུའི་རྩེ་མོར་ཧཱུྃ་སྔོན་པོ་མགོ་གྱེན་ལ་བསྟན་པའི་རྩེ་མོར་ཐིག་ལེ་དམར་པོ་སྲན་མ་ཙམ་གཅིག་བསམ་ལ་ཧཱུྃ་གི་ནུས་པས་ནོར་བུ་ནང་དུ་བཟློག་པ་བཞིན་དུ་བྱུང་སྙིང་ཁར་སླེབས། སྨད་ཀྱི་འདོད་ཆགས་སངས་དག་པར་བསམ། དབུ་མའི་ཡར་རྩེ་སྤྱི་བོར་ཧཾ་དཀར་པོ་མགོ་ཐུར་བསྟན་གྱིས་རྩེ་ལ་ཐིག་ལེ་དཀར་པོ་སྲན་མ་ཙམ་བསམ་པ་དབུ་མ་ན་མར་ལ་བྱུང་། སྙིང་ཁར་སླེབས་པས་གཏི་མུག་དག་པར་བསམ།

བུམ་དབང་དང་འབྲེལ་བའི་དབང་བཞི་ཡན་ཆད་བསྐུར་བ་ནི། སྤྱིའི་འཇུག་སྒོ་རྒྱུད་སྡེ་ལྔའི་དབང་ཡིན། དེ་ནས་མཎྜལ་གསུམ་པའི་བསྐྱེད་ཆོག་བཞིན། བདག་བསྐྱེད་མདུན་བསྐྱེད་བསྒོམ་བཟླས་མཆོད་གཏོར་རྣམས་བྱས། དབང་གི་སྔོན་འགྲོའི་ཆོས་རྣམས་གོང་བཞིན་རྒྱུན་བཤགས་ཡན་ཆད་དྲུག་པོ་རྒྱས་པར་གྲུབ་ན་ལེགས། ཅི་སྟེ་མ་གྲུབ་ན་སྐྱབས་འགྲོ་སེམས་བསྐྱེད་ཙམ་བྱས་ནས་གོང་གི་ཏིང་ངེ་འཛིན་གྱི་དབང་གི་གདམས་ངག་དབང་དང་འབྲེལ་བ་ནི་སྒྱུ་ལུས་ཀྱི་སྒོས་དབང་། དབང་གདམས་ངག་དང་འབྲེལ་བ་གོང་མའི་དམིགས་པ་ནི། གའུ་མའི་སྒོས་དབང་། དབང་གདམས་ངག་དང་འབྲེལ་བ་འོག་མའི་སྟོང་པས་གཞི་བྱས་རིག་པ་སྣ་ཚོགས་སུ་འཆར་བ། འོད་གསལ་གྱི་སྒོས་དབང་།
