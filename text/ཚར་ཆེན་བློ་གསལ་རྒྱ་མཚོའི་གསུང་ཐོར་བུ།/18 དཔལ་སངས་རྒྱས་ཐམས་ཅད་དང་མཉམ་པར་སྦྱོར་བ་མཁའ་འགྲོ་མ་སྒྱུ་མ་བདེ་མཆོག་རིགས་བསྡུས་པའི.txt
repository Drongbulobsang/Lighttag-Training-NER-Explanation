བཀྲ་ཤིས་འདོན་ཞིང་བསྐོར་བར་བྱ། ། ཤཔྟ༞ ཞེས་པས་ཉེར་སྤྱོད། སྒེག་སོགས་བརྒྱད། འདོད་ཡོན་རིན་ཆེན་བདུན། ཡིད་བྱུང་རྣམས་སྒྲུབ་ཐབས་ལྟར་འབུལ། མཆོད་བསྟོད་ཕྱོགས་བཅུ་དུས་གསུམ་གྱི། སོགས་ནས། དྲིན་ཅན་རྩ་བའི་བླ་མའི་ཞལ་དུ་ཨོཾ་ཨཱ་ཧཱུྃ། བཅོམ་ལྡན་འདས་ཀུན་ཏུ་བཟང་པོ༞ རྒྱལ་བ་རིགས་ལྔ༞ གསང་བདག་ཕྱག་ན་རྡོ་རྗེ༞ ལི་ཙ་བི་དྲི་མ་མེད་པ༞ རྒྱལ་པོ་རབ་གསལ་ཟླ་བ༞ རྒྱལ་བུ་ཨུ་པ་རཱ་ཛ༞ གྲུབ་ཆེན་ཀུ་ཀུ་རཱ་ཛ༞ མུ་ཏི་ཏ་ཉི་ཟླ་རྡོ་རྗེ༞ བ་དྷུ་ཏ་མུ་ཁི༞ དཔལ་སངས་རྒྱས་གསང་བ༞ པཎ་ཆེན་བི་མ་ལ་མི་ཏྲ༞ རྨ་ཨཱ་ཙརྻ་རིན་ཆེན་མཆོག༞ གྱེ་རེ་ཆོས་སྐྱོང༞ ཙུ་རུ་རིན་ཆེན་གཞོན་ནུ༞ གོ་འོལ་ཚུལ་ཁྲིམས་འབྱུང་གནས༞ སྟག་རྩེ་འཇམ་དཔལ༞ བྱང་ཆུབ་འོད༞ བློ་གྲོས་རྒྱལ་མཚན༞ རོང་པཎ་རིན་ཆེན་ཚུལ་ཁྲིམས༞ རོང་ཟོམ་ཆོས་ཀྱི་བཟང་པོ༞ ཡཾ་ཤུད་བློ་ཆེན༞ མགོ་ཟླ་བ་འབར༞ མགོ་འབུམ་གྲགས༞ མགོ་དངོས་གྲུབ་རྒྱལ་མཚན༞ མགོ་དགེ་སློང་ཆོས་རྡོར༞ མགོ་དངོས་གྲུབ་རྒྱལ་མཚན༞ དབུས་པ་སངས་རྒྱས་འབུམ༞ ཀུན་མཁྱེན་ཆོས་སྐུ་འོད་ཟེར༞ བླ་མ་མི་བསྐྱོད་རྡོ་རྗེ༞ བྱང་ཆེན་པ་བསོད་ནམས་འཕེལ༞ བག་སྟོན་གཞོན་ནུ་ཚུལ༞ བླ་མ་ཚུལ་ཁྲིམས་རྒྱལ་མཚན༞ བླ་ཆེན་བསོད་ནམས་དཔལ༞ འཇམ་དབྱངས་ཀུན་དགའ་བློ་གྲོས༞ ཆོས་རྗེ་ནམ་མཁའ་ཚུལ་ཁྲིམས༞ རོང་པོ་བློ་བཟང༞ རྡོ་རྗེ་འཆང་མཁྱེན་རབ་ཆོས་རྗེ༞ སློབ་དཔོན་སཾ་གྷ་ཤྲཱི༞ སློབ་དཔོན་ཚུལ་ཁྲིམས་འོད་ཟེར༞ སློབ་དཔོན་བསོད་ནམས་དཔལ་འབྱོར༞ འཇམ་དབྱངས་རིན་ཆེན་བཀྲ་ཤིས་རྒྱལ་མཚན་དཔལ་བཟང་པོའི་ཞལ་སྔ་ནས་རྣམས་ཀྱི་ཞལ་དུ་ཨོཾ་ཨཱཿཧཱུྃ། གཞན་ཡང་དབང་བསྐུར་སོགས་དང་ལྷ་སོ་སོའི་སྔགས་ཀྱི་མཐར་འབྲུ་གསུམ་བཏགས་པ་དང་།
