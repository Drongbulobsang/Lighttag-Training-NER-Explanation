བྱང་ཆུབ་ཀྱི་ཕྱོགས། ། ཕྱོགས་བྱང་ཕྱོགས་ཤུགས་ལ་བསྟན་་་་་་རྡོ་རྗེ་མཁའ་འགྲོ་་་་་་མིང་དོན་གཉིས་པའི་ས་ལ་བསྟོད། ས་གསུམ་པ་ལ་བསྟོད་པ་ལ་དོན་དགུ་ལས་ཡུལ་་་་་་། ཀ་མ་རུ་ནི་ཤར་ན། མཆན་ཁུང་་་་་་ཀཱ་མ་རུ་པར་་་་་་རྩ་སྐྱོན་མ་་་་་་ལས་མིག་་་་་་བྱང་སེམས་འབབ་ཅིང་། ། ཨོ་ཊེར་ནི་ལྷོ་ན་་་་་་ཨོ་་་་་་ཡུལ་་་་་་ནུ་མ་གཉིས་ཊེར་་་་་་རྩ་མཇུག་མ་་་་་་ལས་མཁྲིས་་་་་་པ་་་་་་བྱང་སེམས་འབབ་་་་་་སྦྱངས་གཞི་ས་ཡི་ཁམས། ། ས་བོན་་་་་་ཨཱུ་སྐྱེས་ལྷ་མོ་་་་་་པུཀྐ་སིས་སྦྱོང་རྫུ་འཕྲུལ་གྱི་རྐང་པ་འདུན་པ་དང་མོས་པ་གཉིས། ། ཕྱོགས་་་་་་བྱང་ཤར་་་་་་ཐུན་མོང་གིས་མིང་འོད་བྱེད་་་་་་ཐུན་མོང་མ་ཡིན་པའི་མིང་ཞིང་ལ་ཕྱག་འཚལ་ལོ། ས་བཞི་པ་བསྟོད་པ་ལ་དགུ་ལས་ཡུལ་་་་་་། ཏྲི་་་་་་ལྟེ་བ་་་་་་ཏྲི་ཤ་ཀུ་ནོར་ལྷོ་ནུབ་ན་ཤ་ཀུ་ནོར་་་་་་རྩ་མ་མོ་གློ་བ་་་་་་བྱང་སེམས་འབབ།
ཡུལ་་་་་་། ཀོ་་་་་་ཀོ་ས་ལ་ནི་ལྷོ་ཕྱོགས་ན་སལ་མཚན་་་་་་རྩ་ལས་མོ་རྒྱུ་་་་་་བྱང་སེམས་མ་ཆུ་་་་་་སྦྱང་གཞི་ཡི་ཁམས། ས་བོན་་་་་་། རྀ་སྐྱེས་་་་་་ལྷ་མོ་ཤ་ཝ་རིས་སྦྱོང་རྫུ་འཕྲུལ་་་་་་བྱང་ཆུབ་ཕྱོགས་གཉིས། ཕྱོགས་་་་་་། ཤར་ལྷོ་་་་་་མིང་དོན་ཞིང་འོད་འཕྲོར་་་་་་ཐུན་མོངས་མིན་པའི་མིང་་་་་་ཐུན་མོང་བའི་མིང་ཕྱག་འཚལ་ལོ། ས་ལྔ་པ་ལ་བསྟོད་པ་ལ་དོན་དགུ་ལས་ཡུལ་་་་་་། ཀཱ་ཀ་ལིང་ཀར་ནི་ལྷོ་ན་སིལ་སྦྱིན་་་་་་རྩ་ཆུ་རྒྱུས་་་་་་བྱང་སེམས་འབབ། ། ཡུལ་མགྲིན་པར་་་་་་ལམ་པར་་་་་་ལམ་པ་ཀ་ནི་ཤར་ན་ཚ་་་་་་རྩ་བ་ལྟོ་བ་འབབ་་་་་་བྱང་སེམས་མ་མེ།་་་་་ སྦྱང་གཞི་བཞི།

པྲེ་ཏར་མཚན་མ་རྩ་་་་་་ཤིན་ཏུ་གཟུགས་ཅན་་་་་་མ་ལས་བད་ཀན་་་་་་བྱང་སེམས་འབབ། ། ཡུལ་གྲི་ཧ་དེ་བ་བྱང་ན་་་་་་ཁྱིམ་གྱི་ལྷར་ནི་་་་་་རྩ་སྤྱི་མ་ལས་བྱང་སེམས་་་་་་རྣག་འབབ་་་་་་སྦྱང་གཞི་གཟུགས། ། ས་བོན་་་་་་ལཱྀ་སྐྱེས་་་་་་ལྷ་མོ་གཽ་རིས་སྦྱོང་མཛད་དབང་པོ་་་་་་བྱང་ཆུབ་ཀྱི་ཕྱོགས་སྟོབས། ། ཕྱོགས་་་་་་ཤར་ཕྱོགས་འདུ་བ་་་་་་མིང་དོན་རིང་དུ་སོང་བར་བསྟོད། ། ས་བརྒྱད་པ་ལ་བསྟོད་པ་ལ་དོན་དགུ་ལས་ཡུལ་་་་་་སཽ་རཥྚར་ནི་་་་་་སཽ་རཥྚ་ནི་ནུབ་ན་རྩ་རྒྱུ་སྦྱིན་་་་་བྱང་སེམས་ཁྲག་འབབ་ཅིང་། ཡུལ་སུ་ཝརྣ་ནི་ལྷོ་ནུབ་ན་སྦྱིན་པ་གཉིས་་་་་་། གསེར་གླིང་་་་་་རྩ་སྦྱོར་བྲལ་་་་་་མ་ལས་རྔུལ་་་་་་བྱང་སེམས་འབབ་་་་་་སྦྱང་གཞི་སྒྲ་ཡི་ཁམས། ། ས་བོན་་་་་་། ཨེ་སྐྱེས་་་་་་ལྷ་མོ་ཙཽ་རིས་སྦྱོང་མཛད་་་་་་བརྩོན་འགྲུས་དང་དྲན་པའི་སྟོབས་་་་་་ཕྱོགས་གཉིས་ལྷོར། མིང་དོན་དགུ་་་་་་།

ཡོད་དེ་་་་་་ལྷན་སྐྱེས་་་་་་ཀྱིས་ལྷན་སྐྱེས་ཨཱཿ་་་་་ཚེག་དྲག་ཅན་གྱི་ཆ་དང་སྟེ་བདུད་བྲལ་མ།་་་་་ འི་ཆ་ལ་སོགས་པའི་སྦྱང་གཞི། ། སྟོང་་་་་་པའི་ངོ་བོ་ལ་སྟོང་་་་་་པའི་ངོ་བོར་གྱུར་པའི་ལྷ་མོ་རྣམས་སྦྱོང་བྱེད་དུ་བཞག་་་་་་ནས་སྦྱོང་ཞེས་དོན་མེད་་་་་་ཅིང་དེ།་་་་་ བས་ན། ། ཤེས་རབ་་་་་་ཀྱི་ངོ་བོར་གྱུར་པའི་སྦྱང་གཞི་རྣམས་ཐབས་བདག་་་་་་ཉིད་དུ་གྱུར་པའི་ཧེ་རུ་ཀ་ཡི་དངོས། ། དེ་དག་གི་གྲུབ་དོན་བསྡུས་པ་ནི་་་་་་དེས་ན་རྡོ་རྗེ་ཐེག་པའི་ཟབ་དོན་་་་་་གཞི་ལམ་འབྲས་བུའི་ཀུན། ། ཐབས་དང་ཤེས་རབ་གཉིས་སུ་མེད་པ་་་་་ར་གྱུར་ལ། ། ཁོ་བོའི་་་་་་རྗེ་བཙུན་སེམས་ཉིད་མཆོག་ཏུ་དད་པ་ཡིས། ཟབ་མོ་་་་་་། རྡོ་རྗེ་ཐེག་་་་་་པ་བརྗོད་བྱ་རྗོད་བྱེད་ཀུན་ལ་གུས་པར་ཕྱག་འཚལ་ལོ། །
