མང་ཐོས་ཀླུ་སྒྲུབ་རྒྱ་མཚོའི་གསུང་སྐོར།
པོད ༤༦
༄༅། །མང་ཐོས་ཀླུ་སྒྲུབ་རྒྱ་མཚོའི་གསུང་སྐོར་དཀར་ཆག།
༄༅། །མང་ཐོས་ཀླུ་སྒྲུབ་རྒྱ་མཚོའི་གསུང་སྐོར་དཀར་ཆག། ༠༠༡ ཟབ་དོན་འདུལ་བའི་དཀའ་བའི་གནས་ལ་དཔྱད་པ་ལུང་རིགས་ནོར་བུའི་ཐེམ་སྐས། ༠༠༢ ཚིགས་གསུམ་རིམ་ཅན་དུ་སྒྲུབ་པའི་རྣམ་པར་བཞག་པ། ༠༠༣ སྐྱབས་འགྲོའི་སྡོམ་པ་དང་དགེ་བསྙེན་འབོགས་པའི་ཆོ་ག་སྙིང་པོ་རབ་གསལ། ༠༠༤ བསྙེན་གནས་ཆོ་གའི་ལག་ལེན་གསལ་བའི་སྒྲོན་མེ། ༠༠༥ འདུལ་བའི་གནས་ལ་དོགས་པ་དཔྱོད་པའི་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་མཁས་པའི་གཏམ་བྱ་གསང་བ་སྒོ་འབྱེད། ༠༠༦ འདུལ་བ་ལུང་ལས་འཕྲོས་པའི་དོགས་གནས་གསུང་རྒྱུན་ཉེར་མཁོ། ༠༠༧ དྲིས་ལན་གནས་སྟོན་རབ་གསལ། ༠༠༨ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་འདུལ་བ་སྨན་བཞིའི་ཟུར་བཀོལ་གཞན་ཕན་སྙིང་པོ་མཁས་པའི་གཏམ་བྱ་རིན་ཆེན་ཕྲེང་བ། ༠༠༩ ཆོས་མངོན་པ་མཛོད་ཀྱི་སྤྱི་དོན་ཤེས་བྱ་རབ་གསལ།
༠༡༠ རྒྱུ་འབྲས་རྐྱེན་གསུམ་གྱི་དཀའ་འགྲེལ་བློ་གསལ་མིག་འབྱེད། ༠༡༡ ཆོས་མངོན་པ་མཛོད་ཀྱི་སྤང་བྱ་སྤོང་ཚུལ་གྱི་དཀའ་འགྲེལ་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་མཁས་པའི་གཏམ་བྱ་གསང་བ་སྒོ་འབྱེད། ༠༡༢ ཕྱོགས་ཆོས་འཁོར་ལོའི་རྣམ་བཤད་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་གཞན་དང་ཐུན་མོང་མ་ཡིན་པའི་ཁྱད་ཆོས། ༠༡༣ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་ཚད་མ་གྲུབ་པའི་གཞུང་འགྲེལ་མཐའ་དཔྱོད་དོགས་དཔྱོད་དང་བཅས་པ། ༠༡༤ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་གཏན་ཚིགས་འཁོར་ལོའི་གཞུང་གསུམ་གསལ་བྱེད། ༠༡༥ ཡིད་མངོན་གྱི་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་བློ་གསལ་མིག་འབྱེད་རིགས་པའི་ཐུར་མ། ༠༡༦ ཚད་འབྲས་གཉིས་པའི་གསུང་རྒྱུན་ནོ། ༠༡༧ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ཟ་མ་ཏོག་ཁ་ཕྱེ་བ་ལས་རང་ཉིད་རྒྱས་བཤད་ཀྱི་སྐབས། ༠༡༨ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས། མ་བསལ་བའི་རྣམ་བཤད།
གྲགས་བསལ་གངྒཱའི་ཆུ་རྒྱུན། ༠༡༩ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་ཡན་ལག་རྫས་འགོག་གི་རྣམ་པར་བཤད་པ་གངྒཱའི་ཆུ་རྒྱུན། ༠༢༠ ལེགས་བཤད་ཉི་མའི་འོད་ཟེར་ལས་སྒྲུབ་ངག་གི་རྣམ་བཤད་བདུད་རྩིའི་ཟ་མ་ཏོག་ཁ་ཕྱེ་བ། ༠༢༡ འགོད་ཡུལ་ཕྱི་རྒོལ་གྱི་རྣམ་བཤད་གཞན་གྱི་སྤྱོད་ཡུལ་མིན་པའི་གནད་ཀྱི་སྙིང་པོ། ༠༢༢ ཚད་མའི་བསྟན་བཅོས་ལ་མ་གོ་བ་བརྒྱ་རྩ། ༠༢༣ རྣམ་འགྲེལ་སྔ་ལྡང་གི་སྦྱོར་བ། ༠༢༤ རིགས་ལམ་སྤྱིའི་རྣམ་གཞག་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར། ༠༢༥ ཚད་མ་རིགས་པའི་གཏེར་གྱི་དཀའ་འགྲེལ་གནད་ཀྱི་སྙིང་པོ་གསལ་བྱེད་ལས་རབ་བྱེད་དང་པོའི་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར། ༠༢༦ རབ་བྱེད་གཉིས་པའི་རྣམ་བཤད་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར། ༠༢༧ སྤྱི་དང་བྱེ་བྲག་བརྟག་པ་སྟེ་རབ་བྱེད་གསུམ་པའི་རྣམ་པར་བཤད་པ།
༠༢༨ གཞན་སེལ་བརྟག་པའི་རབ་བྱེད་བཞི་པའི་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར། ༠༢༩ འབྲེལ་པ་བརྟག་པའི་རབ་ཏུ་བྱེད་པ་སྟེ། རབ་ཏུ་བྱེད་པ་དྲུག་པའི་རྣམ་པར་བཤད་པ། ༠༣༠ རབ་བྱེད་བདུན་པ་འགལ་བ་བརྟག་པའི་རབ་ཏུ་བྱེད་པ། ༠༣༡ ཚད་མ་རིགས་པའི་གཏེར་གྱི་དཀའ་འགྲེལ་གནད་ཀྱི་སྙིང་པོ་གསལ་བྱེད་ལས་མཚན་ཉིད་བརྟག་པ་སྟེ་རབ་བྱེད་བརྒྱད་པའི་རྣམ་པར་བཤད་པའོ།། ༠༣༢ རབ་བྱེད་ལྔ་པའི་འགྲེལ་པ་ཆད་པར་བརྗོད་བྱ་རྗོད་བྱེད་ཀྱི་རྣམ་གཞག་ཟུར་དུ་བྱུང་བ། ༠༣༣ རྨི་ལམ་གྱི་རྩོད་གཞི། ༠༣༤ འཕགས་པ་ཀླུ་སྒྲུབ་ལ་རྟེན་འབྲེལ་གྱི་སྒོ་ནས་རྟོགས་པ་བརྗོད་པ་ལྟ་བའི་འདོད་འཇོ་དད་པའི་སྙེ་མ། ༠༣༥ འཕགས་པ་ཀླུ་སྒྲུབ་ཞབས་ཀྱི་ཐུན་མོང་མ་ཡིན་པའི་ལེགས་བཤད་ཀྱི་གསང་གཏམ་སྙན་པའི་སྒྲ་སྒྲོག ༠༣༦ སྡོམ་གསུམ་རབ་དབྱེའི་དཀའ་འགྲེལ་སྦས་དོན་གནད་ཀྱི་སྙིང་པོ་གསལ་བྱེད་ལས།
ཕྱག་ཆེན་རྩོད་སྤོང་སྐབས་ཀྱི་ལེགས་བཤད་ཉི་མའི་འོད་ཟེར། ༠༣༧ ཡང་ལན་མཁས་པའི་མིག་ཐུར་སྤྱན་དབྱེའི་ཆོ་ག ༠༣༨ གསུང་ངག་རིན་པོ་ཆེའི་ཁོག་ཕུབ་ཀྱི་རྣམ་པར་བཤད་པའི་ཟིན་བྲིས་གནད་ཀྱི་བེ་བུམ་ཟབ་མོ། ༠༣༩ གཞུང་བཤད་ཟིན་བྲིས་རྩོམ་འཕྲོ་ཅན། ༠༤༠ ལམ་ཟབ་སྦས་དོན་སྙིང་པ། ༠༤༡ རྒྱལ་བ་སྲས་བརྒྱད་བསྟོད་འཕྲིན་དགེ་ལེགས་ཀུན་འབྱུང་། ༠༤༢ འདྲེན་མཆོག་ལ་བསྔགས་པའི་བསྟོད་དབྱངས་འཆི་མེད་རྔ་སྒྲ། ༠༤༣ རྣམ་སྲས་མདུང་དམར་ཅན་གྱི་བསྐང་བཤགས་དངོས་གྲུབ་འབྱུང་གནས། ༠༤༤ སྡོམ་གསུམ་ཁ་སྐོང་གི་རྣམ་བཤད་ལེགས་པར་བཤད་པ་ནོར་བུའི་ཕྲེང་བ། ༠༤༥ བསྟན་རྩིས་ཆོས་འབྱུང་གསལ་བའི་ཉིན་བྱེད་ལྷག་བསམ་རབ་དཀར།
