རྒྱུད་སྡེ་རིན་པོ་ཆེ་དེ་དག་ལ་བརྟེན་ནས་དམ་པ་ཐམས་ཅད་འདྲེན་པ་འཇིག་རྟེན་གསུམ་གྱི་རྡོ་རྗེས་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྒྲུབ་པའི་ཐབས་ཀྱི་གཞུང་ཆེ་ཆུང་མཛད་པ་ད་ལྟ་སྒྲུབ་ཐབས་རྒྱ་མཚོའི་རྒྱ་གཞུང་ན་བཞུགས་པ་དང༌། ཇོ་བོ་ཆེན་པོ་རྗེ་ནས་བརྒྱུད་པའི་གདམས་ངག་ཐོར་བུར་བཞུགས་པ་རྣམས་མཆིམས་ནམ་མཁའ་གྲགས་ཀྱིས་ཕྱོགས་གཅིག་ཏུ་བསྡེབས་པ་སྣར་ཐང་བརྒྱ་རྩར་གྲགས་པའི་བསྟན་བཅོས་ཆེན་པོ་སོགས་སུའང་བཞུགས་པ་རྣམས་ལ་བརྟེན་ནས་ངོར་ཆེན་ཐམས་ཅད་མཁྱེན་པ་དཀོན་མཆོག་ལྷུན་གྲུབ་པས་མཛད་པའི་སྒྲུབ་ཐབས་རྗེས་གནང་དང་བཅས་པའི་ཡི་གེ་དང༌། པཎ་ཆེན་བྲང་ཏེ་པས་མཛད་པའི་སྒྲུབ་ཐབས་རྒྱ་མཚོའི་བསྟན་བཅོས་ཀྱི་དཀྱུས་སུ་བཞུགས་པ་དང༌། ཡང་བླ་ཆེན་ཤཱཀྱ་དཔལ་བས་ཞལ་གཟིགས་པའི་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྒྲུབ་ཐབས་རྗེས་གནང་མེ་དབང་དང་བཅས་པ་དང༌། སྣར་ཐང་བརྒྱ་རྩའི་བོད་གཞུང་རྣམས་སུ་བཞུགས་པ་རྣམས་ནི། རྒྱུད་གཞུང་གོང་དུ་སྨོས་པ་རྣམས་ཁོ་ནར་ཡིན་ཡང་ཕྱག་བཞེས་དང་ཚིག་ཚོགས་རྒྱས་བསྡུས་ཅུང་ཟད་ཙམ་དང༌། བརྒྱུད་རིམ་གྱི་ཁྱད་པར་དབྱེ་བ་ཙམ་ལས་དོན་ལ་ཁྱད་པར་ཆེར་མ་མཆིས་པས། བླ་མ་དམ་པ་འགའ་ཞིག་ཕན་ཚུན་མི་གསལ་བ་གསལ་བར་བྱེད་པའི་ཚུལ་གྱིས་མེ་དམིགས་བཞི་ལ་སོགས་པ་ཕན་ཚུན་རྒྱས་འགེབས་མཛད་པའི་ཞལ་ཤེས་ཀྱང་མཆིས་སོ། ། འདིར་སྐབས་སུ་བབས་པ་རྗེ་ཐམས་ཅད་མཁྱེན་པ་དཀོན་མཆོག་ལྷུན་གྲུབ་པའི་ཡིག་ཆ་འདི་ཉིད་ཀྱི་སྟེང་ནས་ཉམས་སུ་ལེན་པའི་ཚུལ་གྱི་བསྒོམ་བྱའི་གོ་དོན་དང་ཞལ་རྒྱུན་དམིགས་རིམ་རྣམས་དོན་གཉེར་དག་གི་ངོར་འདྲི་བ་ལ་གསུམ་སྟེ། མངོན་པར་རྟོགས་པའི་སྔོན་འགྲོ་དང༌།

ཞེས་མུ་སྟེགས་བྱེད་ལས་ཁྱད་པར་དུ་འགྱུར་བའི་ཕྱིར་ཐོག་མར་དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་འགྲོ་དགོས་པས། རང་གང་དུ་གནས་པའི་ཞིང་འདི་ཐ་མལ་པ་མ་ཡིན་པ་ཡེ་ཤེས་ཀྱི་རང་སྣང་ལས་གྲུབ་པའི་བཻ་ཌཱུརྻའི་ས་གཞི་ལྟ་བུ་དྭངས་གསལ་འོད་ཀྱི་རང་བཞིན་ཅན་དུ་རང་ཉིད་འཁོད་པའི་རང་གི་མདུན་གྱི་ནམ་མཁར་སེང་གེས་བཏེག་པའི་རིན་པོ་ཆེའི་ཁྲི་པདྨ་ཟླ་བ་དང་བཅས་པའི་སྟེང་དུ། རང་གི་རྩ་བའི་བླ་མ་ཞེས་པའི་དོན་ནི་རྡོ་རྗེ་ཐེག་པའི་ལམ་གྱི་རྩ་བ་ཀྱེ་རྡོ་རྗེ་ལ་སོགས་པ་རྣལ་འབྱོར་བླ་ན་མེད་པའི་དཀྱིལ་འཁོར་ཆེན་པོ་ཞིག་ཏུ་སྨིན་བྱེད་ཀྱི་དབང་བསྐུར་བ་མཚན་ཉིད་དང་ལྡན་པ་ཐོབ་པའི་ཡུལ་དེ་ནི་ཐུན་མོང་མ་ཡིན་པའི་སྡོམ་པ་གསུམ་པོ་ཅིག་ཅར་དུ་ཐོབ་པའི་རྩ་བའི་བླ་མ་ཡིན། དེ་ནས་བརྩམས་ཏེ་གསང་སྔགས་རྡོ་རྗེ་ཐེག་པའི་དབང་དང་བྱིན་རླབས་གདམས་ངག་ཟབ་མོའི་ཚིགས་སུ་བཅད་པ་གཅིག་ཚུན་ཆད་ཐོབ་པའི་བླ་མ་རྣམས་དང༌། ཁྱད་པར་ཕྱག་ན་རྡོ་རྗེ་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་རྗེས་སུ་གནང་བ་དང་གདམས་ངག་ལེགས་པར་ཐོབ་པ་རྣམས་དང་བཅས་པ་སྟེ། དེ་ཐམས་ཅད་ལ་རྩ་བའི་བླ་མ་ཞེས་བྱ། དེས་ན་དེ་ཐམས་ཅད་འདུས་པའི་བདག་ཉིད་ཡིན་པ་ལ་རྣམ་པ་ཕྱག་ན་རྡོ་རྗེ་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྐུར་གསལ་གདབ་པ་རྩ་བའི་བླ་མ་ཀུན་འདུས་ཡིན། དེའི་མཐར་མདུན་ནས་བརྩམས་ཏེ་བརྒྱུད་པའི་བླ་མ་རྣམས་ཀྱང་ཕྱག་ན་རྡོ་རྗེའི་རྣམ་པས་གཡས་སྐོར་དུ་གཙོ་བོ་ལ་ཕྱོགས་པའི་ཚུལ་གྱིས་བཞུགས་པ་དང༌། དེའི་ཕྱི་རོལ་རྩ་བའི་བླ་མའི་མདུན་ཕྱོགས་ཀྱི་ཆར་རྒྱུད་སྡེ་བཞིར་གཏོགས་པའི་ཡི་དམ་ཐམས་ཅད་དང༌། གཙོ་བོའི་གཡས་ཕྱོགས་ཀྱི་ཆར་རྫོགས་པའི་སངས་རྒྱས་ལོངས་སྤྱོད་རྫོགས་པ་དང་སྤྲུལ་པའི་སྐུའི་རྣམ་པ་ཅན་མང་པོ་དང༌།

སེམས་ཅན་ཐམས་ཅད་ཀྱིས་དུས་འདི་ནས་བྱང་ཆུབ་སྙིང་པོའི་བར་དུ་སྐྱབས་སུ་འགྲོ་སྙམ་པས་བདག་དང་འགྲོ་བ་སོགས་ནས་དགེ་འདུན་རྣམས་ལ་སྐྱབས་སུ་མཆིའོ་ཞེས་པའི་བར་ཅི་ནུས་སུ་གདོན་ནོ། ། དེ་ནས་བླ་མ་དང་དཀོན་མཆོག་རིན་པོ་ཆེ་སོགས་ནས་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ་ཞེས་པའི་བར་བརྗོད་པས་སྐྱབས་ཡུལ་རྣམས་འོད་དུ་ཞུ་ནས་བདག་དང་སེམས་ཅན་ཐམས་ཅད་ལ་ཐིམ་པས། དཀོན་མཆོག་གི་བྱིན་རླབས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་རྒྱུད་དུ་ཞུགས་པར་བསམ་མོ། ། གཉིས་པ་ནི། བྱམས་པས། སེམས་བསྐྱེད་པ་ནི་གཞན་དོན་ཕྱིར། ། ཡང་དག་རྫོགས་པའི་བྱང་ཆུབ་འདོད། ། ཅེས་གསུངས་པ་ལྟར་ཏེ། དེ་ལའང་ཐོག་མར་སེམས་ཅན་ཐམས་ཅད་ཚེ་རབས་དུ་མར་ཕ་མ་དང་ཕུ་ནུར་གྱུར་པའི་སེམས་ཅན་སྡུག་བསྔལ་ཅན་རྣམས་ལ་དམིགས་ཏེ། ཀྱེ་མ་འཁོར་འདས་གཉིས་སྣང་གི་ཆོས་ཐམས་ཅད་གདོད་མ་ནས་རང་བཞིན་གྱིས་རྣམ་པར་དག་པ་ལྷ་དང་ཡེ་ཤེས་ཀྱི་རྣམ་རོལ་དུ་འདུག་པ་ལ།

དེའང་འདིར་གུར་ལས་འབྱུང་བའི་ཚིགས་སུ་བཅད་པ་གསུམ་ངག་ཏུ་བརྗོད་པས་ཏེ། དཀོན་མཆོག་གསུམ་ལ་སོགས་ཚིག་རྐང་གཉིས་ཀྱིས་སྡིག་པ་བཤགས་པའི་ཡན་ལག་བསྟན་ཏེ། དེ་ལ་སྟོབས་བཞི་ཚང་བའི་ཐབས་ཀྱིས་བཤགས་ན་སྡིག་པ་ཇི་ལྟར་ཆེ་ཡང་འདག་པར་གསུངས་ཏེ། དེའང་རྟེན་གྱི་སྟོབས་ནི་མདུན་གྱི་ནམ་མཁར་བཞུགས་པའི་ཐུན་མོང་མ་ཡིན་པའི་ཡུལ་དཀོན་མཆོག་གསུམ་བཞུགས་པའི་ཚོགས་ཞིང་རྣམས་སོ། ། རྣམ་པར་སུན་འབྱིན་པའི་སྟོབས་ནི་སྔར་རང་གིས་བྱས་པའི་སྡིག་པ་རྣམས་ནི་དུག་ཁོང་དུ་བཏུང་བ་ལྟ་བུས་འགྱོད་སེམས་དྲག་པོས་བཤགས་པའོ། ། གཉེན་པོ་ཀུན་ཏུ་སྤྱོད་པའི་སྟོབས་ནི་ཕྱིན་ཆད་སྡིག་པའི་ལས་དེ་ལྟ་བུ་དང་གཞན་གང་ཡང་རུང་བ་སྲོག་ལ་བབ་ཀྱང་མི་བྱེད་སྙམ་དུ་དམ་འཆའ་བའོ། ། སོར་ཆུད་པའི་སྟོབས་ནི་ད་ལམ་བདག་གིས་ཡུལ་གཉེན་པོ་འདི་དག་གི་མདུན་དུ་སྡིག་པ་བཤགས་པའི་ཐབས་ཀྱི་ཁྱད་པར་འདི་ལྟ་བུ་བྱས་པས་བདག་རྒྱུད་ཀྱི་སྡིག་པ་ཐམས་ཅད་དག་པར་ངེས་སྙམ་པའི་དགའ་བ་བསྒོམ་པ་སྟེ་སྟོབས་བཞིའོ། ། འགྲོ་བའི་དགེ་ལ་རྗེས་ཡི་རང༌། ། ཞེས་པས་དགེ་བ་ལ་ཡི་རང་བའི་ཡན་ལག་བསྟན་ཏེ། ཁམས་གསུམ་གྱི་འགྲོ་བ་སྟེ་འཁོར་བས་བསྡུས་པ་དང༌།
ཉན་ཐོས་དང༌། རང་རྒྱལ་བ། བྱང་ཆུབ་སེམས་དཔའ། རྫོགས་པའི་སངས་རྒྱས་ཏེ་མྱ་ངན་ལས་འདས་པས་བསྡུས་པ་རྣམས་ཀྱིས་དུས་གསུམ་དུ་བསགས་པའི་དགེ་བའི་རྩ་བ་ཐམས་ཅད་ལ་ཕྲག་དོག་མེད་པར་རྗེས་སུ་ཡི་རང་བསྒོམ་པའོ། ། སངས་རྒྱས་བྱང་ཆུབ་ཡིད་ཀྱིས་གཟུང་ཞེས་པས། དོན་དམ་བྱང་ཆུབ་ཀྱི་སེམས་གཟུང་བའི་ཡན་ལག་བསྟན་ཏེ། སངས་རྒྱས་ཐམས་ཅད་ཀྱིས་ཉེ་བར་བསྟན་པའི་དོན་དམ་པའི་བྱང་ཆུབ་ཀྱི་སེམས་སྟོང་པ་ཉིད་ཀྱི་མཚན་ཉིད་ཅན་དེ་བདག་གིས་ཀྱང་བསྒོམ་སྙམ་དུ་དམ་འཆའ་བའོ། ། སངས་རྒྱས་ཆོས་དང་ཚོགས་མཆོག་སོགས་རྐང་པ་གཉིས་ཀྱིས་སྐྱབས་སུ་འགྲོ་བའི་ཡན་ལག་བསྟན་ཏེ། དཀོན་མཆོག་གསུམ་ནི་ཕྱི་ནང་གསང་བའི་དཀོན་མཆོག་གསུམ་སྟེ། སྐུ་གསུམ་གྱི་བདག་ཉིད་ཅན་ནི་སངས་རྒྱས།

བདག་གིས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དུ་བླ་ན་མེད་པ་རྫོགས་པའི་སངས་རྒྱས་ཀྱི་གོ་འཕང་དེ་ཅི་ནས་ཀྱང་ཐོབ་པར་བྱ་སྙམ་དུ་སྨོན་པའོ། ། བྱང་ཆུབ་མཆོག་གི་སེམས་ནི་བསྐྱེད་བགྱིས་སོགས་ཚིག་རྐང་གསུམ་གྱིས་འཇུག་པ་སེམས་བསྐྱེད་པའི་ཡན་ལག་བསྟན་ཏེ། གོ་འཕང་དེའང་རྒྱུ་རྐྱེན་མེད་པར་མི་འབྱུང་བར་འདུག་པས་དེའི་ཆེད་དུ་ཟབ་མོ་རྡོ་རྗེ་ཐེག་པའི་ལམ་ལ་བསླབ་སྙམ་པའོ། ། འགྲོ་ལ་ཕན་ཕྱིར་སངས་རྒྱས་འགྲུབ་པར་ཤོག་ཅེས་པས་ཡོངས་སུ་བསྔོ་བའི་ཡན་ལག་བསྟན་ཏེ། དེ་ལྟར་སྡིག་པ་བཤགས་པ་སོགས་པའི་ཡན་ལག་དྲུག་པོ་རྣམས་བགྱིས་པའི་དགེ་བས་མཚོན་ཏེ་དུས་གསུམ་དུ་བསགས་པའི་དགེ་བའི་རྩ་བ་ལ་བརྟེན་ནས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དུ་བླ་ན་མེད་པ་རྫོགས་པའི་བྱང་ཆུབ་ཐོབ་པར་གྱུར་ཅིག་པ་སྙམ་པ་སྟེ། དེ་དག་ནི་བླ་མ་དང་ལྷ་ལ་བརྟེན་པས་ན་ཡུལ་དག་པ་བརྟེན་པའི་ཚོགས་བསགས་ཞེས་བྱའོ། ། ཚོགས་ཞིང་དེ་དག་མི་དམིགས་པར་བཞག་པའམ་རྒྱུད་ལ་བྱིན་རླབས་འཇུག་པའི་ཕྱིར་ཚོགས་རྟེན་དེ་དག་གི་ཞལ་ནས་རིགས་ཀྱི་བུ་ཁྱོད་གནས་པ་འདིས་གནས་ན་མཆོག་གི་དངོས་གྲུབ་འགྲུབ་པར་འགྱུར་རོ་ཞེས་ཞལ་གྱིས་བཞེས་པའི་གསུང་ལྷང་གིས་བྱོན་ཏེ་འོད་དུ་ཞུ་ནས་རང་གཞན་སེམས་ཅན་ཐམས་ཅད་ལ་ཐིམ་པར་བསམས་པ་གང་མོས་ཏེ་འདིར་ཕྱི་མ་ཉིད་ལེགས་སོ། ། གཉིས་པ་ནི། མགོན་པོ་བྱམས་པས། སེམས་ཅན་ཁམས་ནི་ཚད་མེད་དེ། །

སྡུག་བསྔལ་དང་སྡུག་བསྔལ་གྱི་རྒྱུ་དང་བྲལ་བར་གྱུར་ཅིག་ཅེས་བརྗོད། གསུམ་པ་དགའ་བ་ཚད་མེད་པ་ནི། དམིགས་ཡུལ་དེ་དག་སྡུག་བསྔལ་མེད་པའི་བདེ་བ་དང་མི་འབྲལ་བར་གྱུར་ན་རེ་དགའ་སྙམ་པའི་འདུན་སེམས་སྨོན་པ་དང་འབྲེལ་བས། སྡུག་བསྔལ་མེད་པའི་བདེ་བ་སོགས་བརྗོད་དོ། ། བཞི་པ་བཏང་སྙོམས་ཚད་མེད་ནི། དམིགས་ཡུལ་དེ་དག་རང་ཕྱོགས་ལ་ཆགས་པ་དང་གཞན་ཕྱོགས་ལ་ཞེ་སྡང་གི་འཆིང་བས་བཅིངས་པ་ཁོ་ན་ཡིན་པས། དེ་དག་དང་བྲལ་བའི་བཏང་སྙོམས་ཆེན་པོ་ལ་གནས་པར་འགྱུར་བའི་འདུན་སེམས་སྨོན་པ་དང་བཅས་པས། ཉེ་རིང་ཆགས་སྡང་དང་བཅས་པའི་བཏང་སྙོམས་སོགས་བརྗོད་པ་སྟེ་འདི་རྣམས་ནི་སེམས་ཅན་ལ་དམིགས་པས་ཡུལ་མ་དག་པ་ལ་བརྟེན་པའི་ཚོགས་བསགས་ཞེས་བྱའོ། ། དེ་ཡན་ཆད་ནི་བསོད་ནམས་ཀྱི་ཚོགས་བསགས་པ་སྟེ། འདི་དག་བསྒོམས་པས་ཚེ་རབས་ཐོག་མ་མེད་པ་ནས་ཡུལ་མཆོག་དམན་ཇི་སྙེད་ཅིག་ལ་བརྟེན་ནས་ལས་ངན་བསགས་པ་ཐམས་ཅད་སྦྱོང་བར་བྱེད་ཅིང་འབྲས་བུ་སངས་རྒྱས་ཀྱི་གཟུགས་སྐུ་འགྲུབ་པའི་རྒྱུ་ཡིན་ནོ། །

སེམས་ཙམ་པས་བརྟགས་པའི་རྣམ་པར་རིག་པ་བདེན་པར་འདོད་པ་ལ་སོགས་པ་མཐར་འཛིན་གྱི་རྟོག་པས་བརྟགས་པ་ཙམ་གྱིས་ཀྱང་མ་བསྒྲིབས་པར་འཁོར་འདས་ཀྱི་ཆོས་ཐམས་ཅད་ཀྱི་རྩ་བ་གཉུག་མ་སེམས་ཀྱི་རང་བཞིན་ཡོད་མེད་ཡིན་མིན་ལ་སོགས་པ་སྤྲོས་པའི་མཐའ་ཐམས་ཅད་དང་བྲལ་བར་གཏན་ལ་ཕབ་ནས། ངག་ཏུ། ཨོཾ་ཤཱུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོ྅ཧཾ་ཞེས་བརྗོད་དེ་སྣང་སྲིད་འཁོར་འདས་ཀྱིས་བསྡུས་པའི་ཆོས་ཐམས་ཅད་རང་བཞིན་གྱིས་མ་དམིགས་པ་སྟོང་པ་ཉིད་འོད་གསལ་ནམ་མཁའ་ལྟ་བུར་གྱུར་པར་བསམ་པར་བྱ་སྟེ། སྔགས་དོན་ནི། སྟོང་ཉིད་ཡེ་ཤེས་རྡོ་རྗེའི་རང་བཞིན་གྱི་བདག་ཉིད་ངའོ་ཞེས་པ་སྐྱེ། དབྱིངས་དང་ཡེ་ཤེས་གཉིས་སུ་མེད་པར་མཚན་རྟོག་གིས་མི་ཕྱེད་པའི་རང་བཞིན་ཅན་གྱི་བདག་ཉིད་ཅན་གང་ལ་ངའོ་ཞེས་ཐ་སྙད་ཙམ་དུ་བྱས་པའོ། ། འདི་ལ་ཡེ་ཤེས་ཀྱི་ཚོགས་བསགས་པ་ཞེས་བྱ་སྟེ། འདི་ཉམས་སུ་བླངས་ན་ཚེ་རབས་ཐོག་མ་མེད་པ་ནས་ང་མེད་ལ་ང་དང་བདག་མེད་ལ་བདག་ཏུ་ཞེན་པའི་ཉོན་མོངས་པ་བག་ཆགས་དང་བཅས་པ་སྦྱོང་བར་བྱེད་ཅིང་འབྲས་བུ་སངས་རྒྱས་ཀྱི་ཆོས་སྐུ་འགྲུབ་སྟེ། སློབ་དཔོན་འཕགས་པས། གཟུགས་ཀྱི་སྐུ་ནི་མདོར་བསྡུ་ན། །
རྒྱལ་པོ་བསོད་ནམས་ཚོགས་ལས་འཁྲུངས། ། དེ་བཞིན་དུ་ནི་ཆོས་ཀྱི་སྐུ། ། ཡེ་ཤེས་ཚོགས་ལས་བྱུང་བ་ཡིན། ། ཞེས་སོ། ། དེ་ལྟར་ཚོགས་གཉིས་ཟུང་འཇུག་ཏུ་ཉམས་སུ་བླངས་པས། ཕར་ཕྱིན་པའི་ལུགས་ཀྱིས་བསྐལ་པ་གྲངས་མེད་མང་པོར་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་གིས་བསྡུས་པའི་ཚོགས་གཉིས་བསྒྲུབ་དགོས་པ། སྔགས་ཀྱི་ཐབས་ལ་མཁས་པའི་ཁྱད་པར་གྱིས་གདན་ཐོག་གཅིག་ཏུ་ཉམས་སུ་ལེན་པ་སྟེ། དེ་དག་གིས་ནི་སྒྲུབ་པའི་ཐབས་ཀྱི་སྔོན་དུ་འགྲོ་བའི་རིམ་པ་རྣམས་བསྟན་ཟིན་ཏོ།། །། སྤྱི་དོན་གཉིས་པ་སྒྲུབ་པའི་ཐབས་ཀྱི་དངོས་གཞི་ལ། མཉམ་པར་གཞག་པའི་རྣལ་འབྱོར་དང༌།

དེའི་ནང་དུ་ཡང་ཇི་ལྟར་རིགས་པའི་བར་སྟོང་ཁྱམས་ཀྱི་རྣམ་པར་ཡོད་པའི་རྭ་བ་ཟླུམ་པོས་བསྐོར་བ་ཕྱི་མའི་དཔངས་དང་མཉམ་པའི་ཐད་ནས་གུར་གྱི་ཕྲག་པ་ཕུབས་པ་ལྟ་བུས་དབུས་རྩེ་བར་ཕྱུར་བུར་ཕུབས་པ། དཔེར་ན་སཱཙྪའི་དབྱིབས་སམ་འདོང་ནས་ཀྱིས་ཕྱུར་བུར་གཏམས་པ་ལྟ་བུ་ཡོད་པའི་རྩེ་མོར་གསེར་གྱི་རྡོ་རྗེ་རྩེ་ལྔ་པ་ཕྱེད་པའི་ཏོག་གིས་སྤྲས་པ། ཕྱི་རོལ་ཐམས་ཅད་ནས་ཡེ་ཤེས་ཀྱི་མེ་འོད་ཁ་དོག་ལྔ་ལྡན་དུས་མཐའི་མེ་ལྟར་རབ་ཏུ་འབར་བ། བདུད་དང་ལོག་འདྲེན་རྣམས་ཀྱིས་འཇུག་པ་ལྟ་ཞོག་བལྟར་ཡང་མི་བཟོད་པའི་གཟི་བརྗིད་དང་ལྡན་པས་བསྐོར་བ་དང་བཅས་པའི་སྲུང་འཁོར་དུ་གྱུར་པར་བསྒོམ་མོ། ། དེ་ལྟ་བུའི་སྲུང་བའི་འཁོར་ལོ་བསྒོམ་པའི་དགོས་པ་ནི། གནས་སྐབས་སུ་ཆོས་སྒྲུབ་པའི་བར་ཆད་ཞི་བ་དང༌། མཐར་ཐུག་བདུད་བཞི་ལས་རྣམ་པར་རྒྱལ་བའི་དགོས་པ་དང་རྟེན་འབྲེལ་ལམ་དུ་བྱེད་པའོ། ། དེའི་ནང་གི་དབུས་སུ་ཧྲཱིཿཡིག་ལྗང་གུ་ལས་བྱུང་བའི་སྣ་ཚོགས་པདྨ་འདབ་མ་བརྒྱད་པ་གེ་སར་དང་བཅས་པའི་ལྟེ་བར་ཨཾ་ཡིག་དཀར་པོ་ལས་བྱུང་བའི་ཟླ་བའི་དཀྱིལ་འཁོར་དྭངས་གསལ་ཉ་གང་བ་ལྟ་བུར་བསྒོམ་མོ། ། གཉིས་པ་ནི། རྡོ་རྗེ་གུར་གྱི་རྒྱུད་ལས།

རྡོ་རྗེ་ཧཱུྃ་དང་བཅས་པ་ཡོངས་སུ་གྱུར་པ་ལས་རང་ཉིད་ཁྲོ་བོའི་རྒྱལ་པོ་དཔལ་ཕྱག་ན་རྡོ་རྗེ་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྐུ་ཞལ་ཕྱག་ཆ་ལུགས་ཡོངས་སུ་རྫོགས་པ་སྒྲུབ་ཐབས་ཀྱི་དཀྱུས་སུ་གསལ་བ་ལྟར་སྣང་ལ་རང་བཞིན་མེད་པ་སྒྱུ་མ་ལྟ་བུའི་སྐུ་ཅན་དུ་གསལ་གདབ། དེའང་མངོན་བྱང་ལྔ་དང་སྦྱར་ན། གོང་ཡེ་ཤེས་ཚོགས་བསགས་སྐབས་ཀྱི་སྟོང་ཉིད་བསྒོམ་པའམ། ཡང་ན་སྟོང་པའི་ངང་ལས་རྡོ་རྗེའི་ས་གཞི་རྭ་བ་གུར་བཅས་སུ་བསྒོམས་པས་ནི་ཆོས་ཉིད་སྟོང་པ་ཉིད་མཚོན་པ་ཡིན་ཏེ། རྒྱུད་རྡོ་རྗེ་རྩེ་མོ་ལས། སྲ་ཞིང་སྙིང་པོ་ཁོང་སྟོང་མེད། ། གཅད་དང་གཞིག་པར་བྱ་བ་མེད། ། བསེག་པར་བྱར་མེད་འཇིག་མེད་པ། ། སྟོང་ཉིད་རྡོ་རྗེ་ཞེས་བྱར་བརྗོད། ། ཅེས་གསུངས་པས་སོ། །

ཤེས་རབ་ཆེན་པོས་འཁོར་བའི་མཐར་མ་ལྷུང་བར་མཚོན་ཏེ་རྩེ་དགུ་ནི་སྟེང་འོག་གི་རྭ་བཞི་བཞི་དང་དབུས་རྭ་ཐལ་བྱུང་དུ་ཡོད་པ་ལ་བཤད་དོ། ། གཡོན་པས་སྡིགས་མཛུབ་དང་བཅས་པའི་རྡོ་རྗེའི་ཞགས་པ་འཛིན་པས་ནི་ཞི་བའི་ཐབས་ཀྱིས་འདུལ་བར་མ་ནུས་པའི་གདུག་པ་ཅན་གྱི་རིགས་རྣམས་བསྡིགས་ཤིང་ཚར་བཅད་དེ་སླར་ཡང་སྙིང་རྗེ་ཆེན་པོས་རྗེས་སུ་འཛིན་པར་མཚོན། རྩ་ཕྱག་གཉིས་ཀྱིས་ཐུགས་ཀར་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་ཕྱག་རྒྱ་མཛད་པས་ནི་དབྱིངས་དང་ཡེ་ཤེས་གཉིས་སུ་མེད་པ་ཐུགས་སུ་ཆུད་པས་གཉིས་འཛིན་གྱི་རྟོག་པ་ཐམས་ཅད་ཤུགས་ལ་འཇོམས་པར་མཚོན། ཞལ་གདངས་ཤིང་མཆེ་བ་བཞི་གཙིགས་པས་བདུད་བཞི་འཇིགས་པར་བྱེད་པར་མཚོན། དམར་ལ་ཟླུམ་པའི་སྤྱན་གསུམ་དང་ལྡན་པས་ནི་སྙིང་རྗེ་ཆེན་པོས་སེམས་ཅན་གྱི་ཁམས་ལ་ཉིན་ལན་གསུམ་མཚན་ལན་གསུམ་སྟེ་དུས་དྲུག་ཏུ་གཟིགས་པར་མཚོན། སྨ་ར་དང་སྨིན་མ་དང་དབུ་སྐྲ་ཁམ་སེར་གྱེན་དུ་བརྫེས་པས་ནི་སངས་རྒྱས་ཀྱི་ཡོན་ཏན་ཐམས་ཅད་གོང་ནས་གོང་དུ་རྒྱས་པར་མཚོན། ཐོད་པ་སྐམ་པོ་ལྔས་དབུ་ལ་བརྒྱན་པས་ནི་ཉོན་མོངས་པ་ལྔ་ཡེ་ཤེས་ལྔར་གནས་གྱུར་པར་མཚོན། མི་མགོ་རློན་པ་ལྔ་བཅུའི་དོ་ཤལ་བྱས་པས་ནི་ནང་གི་ཨཱ་ཕྲེང་ཀཱ་ཕྲེང་ལྔ་བཅུ་དག་པར་མཚོན། ཀླུ་ཆེན་པོ་བརྒྱད་ཀྱིས་བརྒྱན་པས་ནི་སྟོབས་ཀྱི་རྒྱུ་ལ་སོགས་པ་ཀླུའི་རྒྱལ་པོ་རྣམས་དམ་ལ་གནས་ཤིང་ཕྲིན་ལས་ཐམས་ཅད་རྗེས་སུ་སྒྲུབ་པར་མཚོན། བརྒྱད་ནི་སྟོབས་ཀྱི་རྒྱུ་སྔོན་པོས་དབུ་སྐྲ་བཅིངས་པ།
དགའ་བོ་དང་ཉེ་དགའ་བོ་སེར་པོས་རྣ་རྒྱན་བྱས་པ། འཇོག་པོ་དམར་པོས་མགུལ་རྒྱན་བྱས་པ། མཐའ་ཡས་དཀར་པོས་ཚངས་སྐུད་བྱས་པ། ནོར་རྒྱས་དཀར་པོས་རྐེད་རྒྱན་བྱས་པ། ཕྱག་རྒྱའི་ཕྱག་གཉིས་ལ་ནི་རིགས་ལྡན་དཀར་པོ་དང༌། ལྷག་མ་གཉིས་ལ་ནི་དུང་སྐྱོང་དཀར་པོས་ལག་གདུབ་བྱས་པ། པདྨ་དང་པདྨ་ཆེན་པོ་དམར་པོས་ནི་ཞབས་གདུབ་བྱས་པའོ། ། སྟག་གི་པགས་པའི་ཤམ་ཐབས་བྱས་པས་ནི་ཁམས་གསུམ་གྱི་དྲེགས་པ་ཅན་ཐམས་ཅད་བྲན་དུ་བཀོལ་བར་མཚོན། དར་སྔོན་པོས་སྟོད་གཡོགས་གསོལ་བ་ནི་རང་ཉིད་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཐུགས་གཉིས་སུ་མེད་པའི་རྡོ་རྗེའི་རིགས་ལས་མ་འཛོལ་བར་མཚོན། ཞབས་གཡས་བསྐུམས་པས་ནི་ཤེས་རབ་ཆེན་པོས་འཁོར་བའི་མཐར་མ་ལྷུང་བར་མཚོན།

གསུམ་པ་སྒྲུབ་པའི་ཡན་ལག་ནི། རང་གི་ཐུགས་ཀའི་ཧཱུྃ་ལས་ཟེར་འཕྲོས་པས་ཡེ་ཤེས་པ་གོང་ཚོགས་ཞིང་གི་སྐབས་ལྟར་མདུན་གྱི་ནམ་མཁར་སྤྱན་འདྲེན་ཏེ། དེའང་གང་ནས་སྤྱན་འདྲེན་པའི་ཞིང་ནི་རང་བཞིན་གྱི་གནས་ཞེས་བྱ་སྟེ། དེའང་རབ་ཏུ་གནས་པའི་རྒྱུད་ལས། དེ་ཕྱིར་ཆོས་དབྱིངས་ཕོ་བྲང་ལྷུན་གྲུབ་ནས། ། ཞེས་འབྱུང་བ་ལྟར་སངས་རྒྱས་རྣམས་ཆོས་ཀྱི་སྐུ་སྤྲོས་བྲལ་ནམ་མཁའི་དབྱིངས་དང་མཉམ་པར་མུ་མཐའ་མེད་པར་ལྷུན་གྱིས་གྲུབ་པའི་ཞིང་དུ་སྙོམས་པར་ཞུགས་པ་ལས་ནམ་ཡང་མི་གཡོ་བའི་ཕྱིར་གཤེགས་བཞུད་ཀྱི་ཀུན་དུ་རྟོག་པ་མི་མངའ་ནའང༌། རྣལ་འབྱོར་པ་ཐབས་ལ་མཁས་པའི་ཁྱད་པར་འོད་ཟེར་གྱིས་བསྐུལ་བས་སྔོན་གྱི་སྨོན་ལམ་རྗེས་སུ་དྲན་པའི་ཚུལ་གྱིས་སེམས་ཅན་གྱི་དོན་དུ་གཤེགས་བཞུད་ཀྱི་རྣམ་པ་ཅིར་ཡང་སྟོན་པའམ། ཡང་ན་ནམ་མཁའི་ཁམས་ཀུན་དུ་ཁྱབ་པར་བཞུགས་པ་ལ། རང་སྒྲིབ་པས་བསྒྲིབས་ཏེ་མ་མཐོང་བ་ལ། འོད་ཟེར་དེས་སྒྲིབ་གཡོགས་བསལ་ནས་མཐོང་བར་བྱས་པ་དེ་ལ་སྤྱན་འདྲེན་པ་ཞེས་པའི་ཐ་སྙད་བྱས་པའོ། །

གྲ་ཧ་ཙ་འཛིན་པ། ཞབས་སྐྱེད་ཨཱུ་ལ་ཤ་བྱིན་པ་ཤཱུ་ནྱ་ཏཱ་སྟོང་པ་ཉིད་དེ། ཀླད་ཀོར་གྱི་མ་ནི་མ་ནོ་ཏ་སྟེ་ཡིད། དེ་ལྟར་ན་གཟུང་འཛིན་གཉིས་ཀྱིས་སྟོང་པའི་ཡིད་ཅེས་པའི་དོན་ཏོ། ། དེ་ལྟ་བུའི་དོན་ཅན་གྱི་ཧཱུྃ་ཡིག་སྔོན་པོ་ལངས་ཐབས་སུ་བསམས་པའི་མཐའ་སྐོར་དུ་ཧཱུྃ་གི་མདུན་ཕྱོགས་ནས་བརྩམས་ཏེ། ཧཱུྃ་བཛྲ་ཕཊ་ཅེས་པས་གཡས་སྐོར་དུ་འཁོར་བ་དང༌། དེའི་ཕྱི་རོལ་དུའང་ཨོཾ་བྷཱུ་ཏ་ཌཱ་མ་ར་སོགས་སྔགས་རིང་གིས་བསྐོར་བ་གཅིག་སྟེ་ཕྱི་སྐོར་ནང་སྐོར་དུ་ཡོད་པ། སྔགས་ཕྲེང་དང་འོད་ཟེར་གྱི་ཁ་དོག་ནི་སྤྱིར་རྒྱུན་གྱི་བཟླས་པ་དང༌། ཞི་བ་ལ་ཁ་དོག་དཀར་པོ་མུ་ཏིག་སྟར་ལ་བརྒྱུས་པ་ལྟ་བུ། རྒྱས་པ་ལ་ཁ་དོག་སེར་པོ་སྒྲོ་ཚུགས་ལྟ་བུ།

ཕྱི་རོལ་དུ་འཕྲོས་པས་འཇིག་རྟེན་གྱི་ཁམས་ཁྱབ་ཅིང་སྣང་བར་བྱས་པས་ནད་རིམས་གདོན་བགེགས་བྱད་ཁ་རྦོད་གཏོང་མི་དང་མི་མ་ཡིན་པའི་གནོད་པ་ཐམས་ཅད་བསྲེགས་ཤིང་སྦྱངས། གནོད་བྱེད་གདོན་བགེགས་རྣམས་ཀྱི་སྡང་བའི་སེམས་དང༌། གདུག་པའི་སེམས་དང༌། བསམ་སྦྱོར་ངན་པའི་མཐུ་དང་ནུས་པ་ཐམས་ཅད་བཅོམ་སྟེ། དེ་རྣམས་བྱམས་བརྩེ་བྱང་ཆུབ་ཀྱི་སེམས་དང་ལྡན་པར་བྱིན་གྱིས་བརླབས་ཏེ། བདག་ཅག་ཆོས་སྒྲུབ་པའི་སྟོང་གྲོགས་བྱེད་པའི་ལས་ཅན་དུ་གྱུར་པར་བསམ། ཡང་དེ་དག་གིས་འདུལ་བར་མ་ནུས་པའི་མ་རུངས་པ་ཐམས་ཅད་གདུལ་བའི་ཕྱིར་བ་སྤུའི་བུ་ག་རྣམས་ནས་རང་དང་འདྲ་བའི་ཁྲོ་ཆུང་མེའི་ཚ་ཚ་ལྟ་བུ་དཔག་ཏུ་མེད་པ་འཕྲོས་པས་གདུག་པ་ཅན་ཐམས་ཅད་ཚར་བཅད་དེ་ལས་ངན་སྡིག་སྒྲིབ་སྦྱངས། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ཀྱི་ཐུགས་རྗེ་དང་བྱིན་རླབས་ནུས་མཐུ་ཐམས་ཅད་བཅོམ་ལྡན་འདས་འབྱུང་པོ་འདུལ་བྱེད་ཀྱི་སྐུ་དང་འོད་ཟེར་གྱི་རྣམ་པ་ཅན་དུ་བསྡུས་ཏེ། རང་གི་ཐུགས་ཀའི་ཧཱུྃ་ལ་ཐིམ་པར་བསམས་ལ་སྔགས་བཟླའོ། ། མདོར་ན་ཕྱག་ན་རྡོ་རྗེ་འདི་ཉིད་ཀྱི་གདུལ་བྱ་ཐུན་མོང་མ་ཡིན་པ་ནི།

ཞེས་འབྱུང་བའི་མན་ངག་ནི་འདི་རྫོགས་རིམ་རླུང་སྦྱོར་གྱི་རྣལ་འབྱོར་ཡིན་པས་ཉམས་སུ་ལེན་པ་ལ་ཐོག་མར་ལུས་གནད་རྣམ་སྣང་གི་ཆོས་བདུན་དང་ལྡན་པ་ལེགས་པར་བཅས་ནས། ཡིད་ལ་ཅི་ཡང་མི་བསམ་པར་སྣ་སྒོ་ནས་རླུང་དྲག་ཏུ་ལན་གསུམ་ཕྱིར་བུས་ལ། རླུང་སྣ་གཡས་གཡོན་གང་ནས་རྒྱུ་བ་ངོས་བཟུང་ནས། ཐོག་མར་ནང་དུ་འཇུག་པ་ཨོཾ་དཀར་པོ། ནང་དུ་གནས་པ་ཨཱཿདམར་པོ། ཕྱིར་འགྲོ་བ་ཧཱུྃ་སྔོན་པོ་ཞེས་དྲན་པར་བྱ། དེ་ལ་ཏིང་ངེ་འཛིན་སྐྱེས་སྐྱེས་ཀྱི་བར་དུ་བསྒོམ། ཏིང་ངེ་འཛིན་སྐྱེས་ན་ནང་དུ་རིང་དུ་རྒྱུས། ནང་དུ་གནས། ཕྱིར་ཐུང་ངུར་རྒྱུས།

སྟོང་པའི་ངང་ལས་ཨ་ལས་བྱུང་བའི་ཡེ་ཤེས་ཀྱི་ཀ་པཱ་ལ་ཡངས་ཤིང་རྒྱ་ཆེ་བའི་ནང་དུ་ཧཱུྃ་ཞུ་བ་ལས་བྱུང་བའི་གཏོར་མ་ཁ་དོག་དྲི་རོ་ནུས་པ་ཕུན་སུམ་ཚོགས་པ་ཟག་པ་མེད་པའི་ཡེ་ཤེས་ཀྱི་བདུད་རྩིའི་རྒྱ་མཚོ་ཆེན་པོར་གྱུར་པར་བསམས་ལ་འབྲུ་གསུམ་ལན་གསུམ་གྱིས་བྱིན་གྱིས་བརླབ། སྐུ་རྟེན་ཡོད་པ་དང་དབྱེར་མེད་དུ་ནམ་མཁར་དྲི་མ་མེད་པའི་ཧཱུྃ་ཡིག་ཡོངས་སུ་གྱུར་པ་ལས་རྡོ་རྗེའི་ཕོ་བྲང་སོགས་ནས་མི་བསྐྱོད་པས་དབུར་བརྒྱན་པའི་བར་བདག་མདུན་གྱི་ཁྱད་པར་ཅུང་ཟད་མ་གཏོགས་སྒྲུབ་ཐབས་ལྟར་ཡན་ལག་བཞི་རྫོགས་སུ་བསྐྱེད། བསྡུ་ན། རང་གི་ཐུགས་ཀའི་ཧཱུྃ་ལས་འོད་ཟེར་འཕྲོས་པས་སྐད་ཅིག་གིས་མགྲོན་རྣམས་སྤྱན་དྲངས། གཉིས་ཀ་ལྟར་ཡང་གཏོར་མགྲོན་གྱི་ལྷའི་ལྗགས་རྡོ་རྗེའི་འོད་ཀྱི་སྦུ་གུ་ཅན་གྱིས་བཅུད་ཡོངས་སུ་ཚིམ་པར་དྲངས་པར་བསམས་ལ། སྔགས་རིང་ཐུང་གི་མཐར་ཨོཾ་ཨ་ཀཱ་རོ་བཏགས་པ་ཅི་རིགས་པས་ཕུལ། མཚན་སྔགས་སྔོན་འགྲོའི་ཨརྒྷཾ་ལ་སོགས་པས་མཆོད། དཔལ་ཆེན་ནམ་མཁའི་མདོག་ལ་སོགས་པས་བསྟོད། དངོས་གྲུབ་གོང་བཞིན་བླངས་ནས། ཡི་གེ་བརྒྱ་པ་ལན་གསུམ་དང༌།

དེ་རྗེས་མཆོད་གཏོར་ཁ་གསོས་ཏེ་མཆོད་པ་བྱིན་རླབས་ནས་བརྩམས་ཏེ་དགོངས་མོའི་བར་གཏོར་བསྔོ་མཇུག་སྡུད་བཅས་བྱ་བ་ནི་གུང་ཐུན་དང༌། དེ་ནས་སྲོད་ཐུན་ལ་འཇུག་པ་སྟེ་དེ་ལྟར་ཐུན་གསུམ་གྱི་རིམ་པས་ཟླ་བ་གཅིག་ཏུ་སྔགས་ཐུང་འབུམ་ཕྲག་བཅུ། སྔགས་རིང་ཁྲི་ཕྲག་བཅུ་ཁ་སྐོང་དུ་བཅས་གྲུབ་པ་དང༌། མཐར་སྔགས་ཀྱི་ནུས་པའམ་འབྲས་བུ་འབྱིན་ཕྱིར་ཞི་བའི་སྲེག་བླུགས་བཅས་གྲུབ་ན་ཤིན་ཏུ་ལེགས་པས། དེའི་ཡིག་ཆ་གཞན་དུ་གསལ་ལོ། ། གསུམ་པ་ནི། དེ་ལྟར་བསྙེན་པ་མཐར་ཕྱིན་པས་འདིའི་སྒོ་ནས་རང་གཞན་གྱི་དོན་དུ་ལས་ཀྱི་ཚོགས་མཐའ་དག་བསྒྲུབ་པ་ལ་དབང་ཞིང༌། གནས་སྐབས་དང་མཐར་ཐུག་གི་ཕན་ཡོན་ནི་འདི་ཉིད་ཀྱི་རྒྱུད་ལས། དཀྱིལ་འཁོར་ཆེན་པོ་འདི་མཐོང་བ་ཙམ་གྱིས་ཁམས་གསུམ་པའི་རྒྱལ་སྲིད་འཐོབ་ཅིང་འདིའི་རྡོ་རྗེ་འཆང་བཟླས་པ་ཙམ་གྱིས་རྡོ་རྗེ་འཆང་དང་མཚུངས་པར་འགྱུར་རོ། ། གལ་ཏེ་མ་གྲུབ་ན་ཡང་གླིང་བཞི་པའི་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོར་འགྱུར་རོ། །
