རྡོ་རྗེ་ཡན་ལག་མེད་པའི་སྒྲུབ་ཐབས་ལ་བཞི། བསྙེན་པ་དང༌། སྒྲུབ་པ་དང༌། སྔགས་ཕན་ཡོན་དང་བཅས་པ་དང༌། ས་བོན་གྱི་ཡི་གེའོ། ། བྱང་ཆུབ་ཀྱི་སེམས་ལ་གཉིས། རིགས་ཐམས་ཅད་ཀྱི་བྱང་ཆུབ་སེམས་ཀྱི་ཡི་གེ་ཧཱུྃ་གི་ཆེ་བ་དང༌། སོ་སོའི་བྱང་ཆུབ་ཀྱི་སེམས་སོ། ། དེ་ལ་གསུམ། ངོ་བོ་ཉིད་དང་འབྲས་བུའི་ཡེ་ཤེས་ལྔའི་རང་བཞིན།
བསྟན་པའི་ཕྱིར་རིགས་ལྔའི་བྱང་ཆུབ་ཀྱི་སེམས་དང༌། ལམ་ཇི་ལྟར་སྒྲུབ་པའི་ཐབས་བསྟན་པའི་ཕྱིར་ཡུམ་ལྔའི་བྱང་ཆུབ་ཀྱི་སེམས་དང༌། འཁོར་ཐམས་ཅད་ཀྱི་བྱང་ཆུབ་ཀྱི་སེམས་སོ། ། དང་པོ་ལ་ལྔ། ཆོས་ཀྱི་དབྱིངས་ཀྱི་ཡེ་ཤེས་དང༌། མེ་ལོང་ལྟ་བུ་དང༌། མཉམ་པ་ཉིད་དང༌། སོ་སོར་རྟོག་པ་དང༌། བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས་ཀྱི་སྒོ་ནས་བསྒོམ་པའོ། ། ཡུམ་གྱི་ལ་ལྔ།
སྣང་བ་སྔོན་དུ་སོང་བའི་སྟོབས་དང༌། སྟོང་པ་སྔོན་དུ་སོང་བའི་སྣང་བ་དང༌། སྲིད་ཞི་མཉམ་པ་དང༌། དམིགས་མེད་བརྗོད་བྲལ་དང༌། སྒྱུ་མ་ལ་བློ་སྦྱངས་ལ་སྟོང་པས་རྒྱས་གདབ་པའོ། ། དཀྱིལ་འཁོར་དང་འབྲེལ་པའི་བྱ་བ་སྣ་ཚོགས་ལ་བཞི། མདའ་གཞུའི་བཀོད་པ་དང༌། རོལ་མོའི་བྱ་བ་དང༌། བདུག་སྤོས་ཀྱི་སྦྱར་ཐབས་དང༌། རྐང་སྟབས་དང་ལག་སྟབས་ཀྱི་བཀོད་པ་མདོར་བསྟན་པའོ། །
སྦྱིན་སྲེག་ལ་ལྔ་སྟེ། ནང་གི་སྦྱིན་སྲེག་དང༌། གཉིས་ཀ་ལ་ཐུན་མོང་དུ་རྫས་མ་འབྱོར་པ་རྣམས་ནི་ཡིད་ཀྱིས་བསམས་ལ་དབུལ་བ་དང༌། ཐུན་མོང་དུ་བསྲེག་རྫས་ཀྱི་སྔགས་དང༌། བསྲེག་རྫས་ཀྱི་ཁྱད་པར་བསྟན་པ་དང༌། ཕྱི་ནང་གི་སྦྱིན་སྲེག་གི་ཕན་ཡོན་བསྟན་པའོ། ། དཀྱིལ་འཁོར་གྱི་ས་གཟུང་བའི་བྱ་བ་ལ་དྲུག རྐང་སྟབས་ཀྱི་བཀོད་པ་དང༌། ལག་སྟབས་ཀྱི་བཀོད་པ་དང༌། འདུག་སྟངས་ཀྱི་བཀོད་པ་དང༌།
བཞིན་གྱི་བསྒྱུར་ཚུལ་དང༌། གླུའི་རབ་ཏུ་དབྱེ་བ་དང༌། དེ་རྣམས་ས་གཟུང་བའི་དུས་སུ་བྱ་བར་བསྟན་པའོ། ། མཆོད་པ་ལ་གཉིས། མེ་ཏོག་ལ་སོགས་པའི་མཆོད་པ་དང༌། རྡོ་རྗེ་དྲིལ་བུ་གཟུང་བའི་སྔགས་བསྟན་པའོ། ། ལེའུ་བཅུ་བཞི་པའོ།། །། ལེའུ་བཅོ་ལྔ་པ་ལ་དགུ། ཧེ་རུ་ཀ་གསུམ་གྱི་རྣལ་འབྱོར་དང༌། ཀུ་རུ་ཀུལླེ་དང༌།
རྣལ་འབྱོར་གྱི་རྒྱུད་ཀྱི་གླེང་གཞི་དང༌། ལས་རྒྱས་པའི་ལྷའི་རྣལ་འབྱོར་དང༌། ཉེ་བའི་རྒྱུ་བསྟན་པ་དང༌། དབང་གི་རྐང་གྲངས་དང༌། དམ་ཚིག་གི་ཚོགས་བསྟན་པ་དང༌། རིགས་ལྔ་བདག་མེད་མ་དང་བཅས་པའི་བསྟོད་པ་དང༌། དངོས་གྲུབ་ཐམས་ཅད་འབྱུང་བའི་རྩ་བ་བླ་མ་ལ་གུས་པར་བྱ་བའོ། ། ལས་རྒྱས་པའི་ལྷའི་རྣལ་འབྱོར་ལ་བཞི། ལྷའི་རྣལ་འབྱོར་བཅུ་བསྟན་པ་དང༌། དེ་དག་གི་སྔགས་བཅུ་བསྟན་པ་དང༌།
ཕན་ཡོན་བཅུ་བསྟན་པ་དང༌། རྡོ་རྗེ་ནག་པོ་ཆེན་པོའི་སྒྲུབ་ཐབས་ཕྲིན་ལས་དང་བཅས་ཏེ་བསྟན་པའོ། ། ཉེ་བའི་རྒྱུ་ལ་བཞི། འབྲས་བུ་འགྲུབ་པའི་ཚུལ་ཐབས་དང་བཅས་པ་དང༌། འགྲུབ་པའི་དུས་དང༌། ཐབས་ཀྱི་ཁྱད་པར་དང༌། འབྲས་བུ་དངོས་སོ། ། དམ་ཚིག་དང་སྡོམ་པ་ལ་བཞི། རྩ་བའི་ལྟུང་བ་དང༌། མི་འབྲལ་བའི་དམ་ཚིག་དང༌།
སེམས་བརྟན་པ་ཐོབ་པའི་དམ་ཚིག་དང༌། ལས་དང་པོ་པའི་དམ་ཚིག་གོ ། དངོས་གྲུབ་ཐམས་ཅད་འབྱུང་བའི་རྩ་བ་བླ་མ་མཉེས་པར་བྱ་བ་ལ་གསུམ། འཐད་པ་དང༌། ཕྱག་རྒྱ་ཆེན་པོ་འགྲུབ་ཚུལ་དང༌། རྗེས་ལ་བསྙེན་བཀུར་བྱ་བའི་ཚུལ་ལོ། ། དང་པོ་ལ་བཞི། སངས་རྒྱས་ཀྱིས་ཀྱང་བསྙེན་བཀུར་བྱེད་པའི་ཕྱིར་དང༌། དེ་མཉེས་ན་མཆོག་གི་དངོས་གྲུབ་འགྲུབ་པའི་ཕྱིར་དང༌། དེ་མཉེས་ན་ཐུན་མོང་གི་དངོས་གྲུབ་འགྲུབ་པའི་ཚུལ་དང༌།
བླ་མ་རྡོ་རྗེ་འཆང་ཉིད་ཀྱིས་སྤྲུལ་པ་ཡིན་པའི་ཕྱིར་རོ། ། གསལ་བའི་སྙིང་པོས་དོན་བསྡུས་མཛད། ། བླ་མས་ལེའུ་བཞི་ཞིག་ཕྱེ། ། དགེ་བསྙེན་གྲགས་པ་རྒྱལ་མཚན་གྱིས། ། ཤིན་ཏུ་གསལ་བར་ཕྱེ་བ་ལགས། ། འཕགས་པ་རྡོ་རྗེ་མཁའ་འགྲོ་མ་རྡོ་རྗེ་གུར་གྱི་སིནྡྷཱརྠ་རྫོགས་སོ།། །། ༈ གུ་རུས་དཔར་རྩ་སྤེལ་བའོ།།
