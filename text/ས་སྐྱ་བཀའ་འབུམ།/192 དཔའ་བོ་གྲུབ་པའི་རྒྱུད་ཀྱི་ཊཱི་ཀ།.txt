ད་དུང་རྒྱུད་ཀྱི་ལྷག་མ་ཡོད་པ་ཡིན་ནོ་གསུང༌། བསྡུས་པའི་དོན་བསྟན་ཟིན་ཏོ།། །། ད་ནི་མན་ངག་གི་གནས་གང་དགོས་པ་རིམ་གྱིས་བརྗོད་པར་བྱ་སྟེ། ཐམས་ཅད་ཀྱང་མ་བསྙེན་པར་ལས་འགྲུབ་པས་ན། གྲུབ་པའི་གསང་སྔགས་ཞེས་བྱ་བར་བཞེད་ཀྱང་སྔོན་དུ་བསྙེན་པ་བྱས་ན་འགྲུབ་པ་ཡིན་པས། གསང་སྔགས་གང་ཟློས་ཀྱང༌། སྔོན་དུ་དཀོན་མཆོག་གསུམ་ལ་དད་ཅིང་གུས་པས་སྐྱབས་སུ་འགྲོ་བ་དང༌། སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དུ་སངས་རྒྱས་ཐོབ་པར་བྱ་ལ། དེའང་སེམས་ཅན་གྱི་དོན་ལ་རག་ལས་པས་ན་གསང་སྔགས་ཀྱི་རྒྱལ་པོ་འདི་དག་གིས་ཕྲིན་ལས་བསྒྲུབས་ལ། སེམས་ཅན་གྱི་དོན་བྱ་སྟེ།

ཅི་འབྱོར་པའི་ཤིང་བརྩིགས། རང་རང་གི་ཐད་སོ་ནས་འབྱུང་བའི་བསྲེག་རྫས་དང་པཱུ་ཛ་མདུན་དུ་བཞག མེ་སྦར་ལ་བདུད་རྩི་འཁྱིལ་པའི་སྔགས་ཀྱིས་བསངས། ཐབ་ཏུ་སྐད་ཅིག་མས་མེ་ལྷ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་ར་ལ་ཞོན་པ་དབྱུ་གུ་དང་གུནྡྷེ་འཛིན་པ་དགུག ཨོཾ་ཨགྣ་ཡེ་སྭཱ་ཧཱ། ཞེས་པས་རྫས་ཕུད་ཀྱི་སྦྱིན་སྲེག་ལན་བདུན་ཙམ་དབུལ། དེ་འབར་བའི་རྣམ་པར་གྱུར་པ་དང་རང་རང་གི་ལྷའམ་དེ་མེད་པ་རྣམས་ལ་ཡི་དམ་གྱི་ལྷ་སྐད་ཅིག་མས་བསྐྱེད། མཆོད་ལ་བསྲེག་རྫས་མང་དུ་ཕུལ་ཏེ་འདོད་པའི་དོན་ལ་གསོལ་བ་བཏབ་ལ་གཤེགས་སུ་གསོལ། འཇིག་རྟེན་པའི་མེ་ལྷ་བསྐྱེད་ལ་ལྷག་མ་སྦྱིན་ནོ། ། འདོད་པའི་དོན་ལ་གསོལ་བ་བཏབ་ལ།

སྐུ་གདུང་དང་བཅས་པའི་མཆོད་རྟེན་ནམ་སངས་རྒྱས་ཤཱཀྱ་ཐུབ་པའི་དྲུང་དུ་པཱུ་ཛ་བཤམས་ལ། དེ་དག་གིས་དཀོན་མཆོག་མཆོད་དེ་གསོ་སྦྱོང་ལ་གནས་པས་བསྙེན་པར་བྱའོ། ། གཞུང་ལས་བསྙེན་པའི་ཆོ་ག་ཁྲི་བཟླས་ལ་ཞེས་འབྱུང་ཡང་འབུམ་ཕྲག་བཞི་བཟླས་ན་གཞུང་ནས་འབྱུང་བའི་ལས་རྣམས་འགྲུབ་པ་ཡིན་ནོ། ། གསང་སྔགས་བཅུ་གཉིས་པའི་སྐབས་སུ་ཡི་གེ་འབུམ་བཟླས་པ་བྱས་ལ་ཞེས་པ་ནི། ཨོཾ་ཨ་ཏི་དེ་སྭཱ་ཧཱ་ཞེས་པ་ཡི་གེ་དྲུག་ཡོད་པས་དྲུག་འབུམ་བསྙེན་པ་བྱས་ལ་ལས་རྣམས་བསྒྲུབས་ཏེ། དེ་བ་དཱཏྟ་ལ་སོགས་པའི་སྔགས་ནི་གང་འདོད་པའི་མིང་འཇུག་པའི་སྐབས་ནི། ཕྱིས་བསྒྲུབ་པའི་དུས་སུ་འཇུག་པ་ཡིན་ནོ། ། གསང་སྔགས་བཅུ་བདུན་པའི་བདག་ཉིད་ཛམྦྷ་ལའི་གཟུགས་སུ་ཞེས་པ་ནི། ཚོགས་གཉིས་བསག་པ་རྒྱས་པའམ། སྐྱབས་འགྲོ་དང་བྱང་ཆུབ་ཀྱི་སེམས་དང༌།

ལྟེ་བར་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཨོཾ་བསམ། དེ་ལས་བྱུང་བའི་ངག་གི་རང་བཞིན་མ་ལུས་པའི་ཕྲེང་བ་ཞེས་བྱ་བ། ཨཱ་ལི་ཀཱ་ལིའམ། ཨོཾ་འབའ་ཞིག ཨོཾ་ལས་སྐྱེས་པར་བསམ་ཞིང༌། ཨོཾ་མམ་ཨཱ་ལི་ཀཱ་ལི་གང་རུང་གཅིག་བཟླ་ཞིང༌། སྔོན་དུ་བསྙེན་པ་བྱའོ། ། དྲུག་པ་ཨོཾ་མམ་བརྫུས་ཏེ་སྐྱེ་བའི་ཚུལ་གྱིས་དཀར་ཆེན་མོ་པདྨ་དང༌། མཆོག་སྦྱིན་གྱི་ཕྱག་རྒྱ་ཅན་མའོ། ། བདུན་པ་ཀརྨ་ལི་ལ་སོགས་པ་བརྗོད་པ་ཙམ་གྱིས་དཔའ་བོ་གྲུབ་པ་གཙོ་འཁོར་ལྔ་བསྒོམ་པར་བྱའོ། །
