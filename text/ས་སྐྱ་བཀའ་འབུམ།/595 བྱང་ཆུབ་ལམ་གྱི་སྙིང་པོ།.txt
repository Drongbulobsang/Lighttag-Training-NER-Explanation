བླ་མེད་རྫོགས་པའི་བྱང་ཆུབ་ཐོབ་པར་ཤོག ། ཅེས་ལན་གསུམ་དུ་བརྗོད་པར་བྱའོ། ། དེ་ནས་བླ་མ་ལ་སོགས་པ་མདུན་དུ་དམིགས་པའི་ཡུལ་དེ་དག་རང་གི་གནས་སུ་གཤེགས་པར་དམིགས་ནས་རང་ཉིད་ཅི་བདེ་བར་སྤྱད་དོ། ། དངོས་གཞིའི་རྣལ་འབྱོར་རོ། ། དོན་གསུམ་པ་རྫོགས་པར་བྱེད་པ་ལམ་གྱི་ཡན་ལག་ནི། བདག་ཉིད་སྡིག་པ་མི་དགེ་བའི་ལས་ལས་ལྡོག་ཅིང༌། གལ་ཏེ་སྡིག་པ་དང༌། ལྟུང་བ་བྱུང་ན་ཕྱོགས་བཅུའི་སངས་རྒྱས་རྗེས་སུ་དྲན་པར་བྱས་ནས་བཤགས་པ་དང༌། ཅི་ནུས་ཀྱིས་དགེ་བའི་རྩ་བ་བསྒྲུབ་པ་ལ་བརྩོན་ཞིང༌། དགེ་བ་གང་བྱས་པ་ཐམས་ཅད་གཞན་གྱི་དོན་དུ་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་བསྔོ་བ་དང༌།
གཞན་དག་ཀྱང་སྡིག་པ་ལས་ལྡོག་ཅིང༌། གལ་ཏེ་སྡིག་པས་གོས་ན་བཤགས་སུ་འཇུག་པ་དང༌། དགེ་བ་ལ་བསྐུལ་ཞིང་བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ་བསྔོར་འཇུག་པ་དང༌། སེམས་ཅན་ཐམས་ཅད་བདེ་བ་དང་ཕྲད་པར་འདོད་པའི་བྱམས་པ། སྡུག་བསྔལ་དང་བྲལ་བར་འདོད་པའི་སྙིང་རྗེ། བདག་དང་སེམས་ཅན་ཐམས་ཅད་མྱུར་དུ་འཚང་རྒྱ་བར་གྱུར་ཅིག་སྙམ་པའི་བྱང་ཆུབ་ཀྱི་སེམས་རྣམས་རྒྱུན་དུ་བསྒོམ་ཞིང༌། ཕྱི་རོལ་ན་ཇི་ལྟར་སྣང་བའི་ཆོས་ཐམས་ཅད་སྒྱུ་མ་ལྟ་བུ་དང༌། ནང་དུ་ཇི་ལྟར་ཉམས་སུ་མྱོང་བའི་སེམས་ཐམས་ཅད་རྨི་ལམ་ལྟ་བུ་དང༌། དེ་ཐམས་ཅད་ཀྱི་རང་བཞིན་གང་གིའང་ངོ་བོར་མ་གྲུབ་པས་སྤྲོས་པ་དང་བྲལ་པ་ནམ་མཁའ་ལྟ་བུར་རྒྱུན་ཏུ་བསམ་མོ། ། ལུས་དང་སེམས་ལ་སྡུག་བསྔལ་མྱོང་བའི་ཚེ།
བདག་གིས་སྔོན་བསགས་པའི་སྡིག་པ་འདག་པར་འགྱུར་སྙམ་པའི་དགའ་བ་བསྒོམ་ཞིང༌། བདག་གིས་འདི་ལྟ་བུའི་སྡུག་བསྔལ་ཉམས་སུ་མྱོང་བ་འདིས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་སྡུག་བསྔལ་ཡང་ཟད་ཅིང༌། ཡོངས་སུ་གཏུགས་པར་གྱུར་ཅིག་སྙམ་དུ་བསྒོམས་ལ་ཡིད་ཞུམ་པ་དང་སྒྱིད་ལུག་པར་མི་བྱའོ། ། ལུས་དང་སེམས་ལ་བདེ་བ་དང༌། དགའ་བ་སྐྱེས་པའི་ཚེ། དགའ་བ་དང་བདེ་བ་འདི་ལྟ་བུ་དང༌། བདེ་བས་ཆེས་ཁྱད་པར་དུ་འཕགས་པའི་དགའ་བདེ་ཐམས་ཅད་ཀྱང་དགེ་བ་སྤྱད་པ་ལས་བྱུང་ངོ༌། ། ཞེས་དགེ་བའི་ཕན་ཡོན་རྗེས་སུ་དྲན་པ་དང༌། དགེ་བ་སྤྱད་པ་དེ་དག་ཀྱང་བླ་མ་དང༌། དཀོན་མཆོག་ལ་བརྟེན་ནས་འབྱུང་ངོ་ཞེས་ཡུལ་གྱི་བཀའ་དྲིན་རྗེས་སུ་དྲན་པར་བྱས་ནས།
སེམས་ཅན་ཐམས་ཅད་འདི་ལྟ་བུ་དང༌། འདི་བས་ཆེས་ལྷག་པའི་དགའ་བ་དང་བདེ་བ་རྒྱུན་མི་འཆད་པ་དང་ལྡན་པར་གྱུར་ཅིག་སྙམ་དུ་བསམས་ནས། བདེ་བའི་གཞི་དེས་ཡིད་ཞེན་པ་དང༌། དྲེགས་པར་མི་འགྱུར་བ་ལ་བསླབ་པར་བྱའོ། ། ལམ་གྱི་ཡན་ལག་གོ ། དེ་ལྟར་ཆོས་གསུམ་ལ་རྟག་ཏུ་གོམས་པའི་རྣལ་འབྱོར་ལ་བརྩོན་པ་ནི་འཇུག་པ་བདེ་ཞིང་ཐབས་སླ་བས་བྱང་ཆུབ་ཀྱི་ལམ་རྣམས་རིམ་གྱིས་བགྲོད་དེ། འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་དགའ་བ་དང༌། བདེ་བ་ཐམས་ཅད་ལ་མངོན་པར་སྤྱད་ནས། མཐར་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར་བ་ཉིད་དོ། ། དེ་ལྟར་ལུང་དང་མན་ངག་ཇི་བཞིན་དུ། །
