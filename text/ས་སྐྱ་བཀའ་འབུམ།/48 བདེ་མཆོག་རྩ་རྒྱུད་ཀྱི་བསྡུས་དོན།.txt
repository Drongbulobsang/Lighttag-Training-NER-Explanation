རྫོགས་པའི་རིམ་པ་དང་སྦྱར་བ་གཉིས་ལས། བསྐྱེད་རིམ་དང་སྦྱར་བ་ལ་ལྔ་ལས་མངོན་པར་རྟོགས་པ་རྫོགས་པ་གཅིག་སྐུ་གསུམ་དུ་སྦྱར་བ་དང༌། གཉིས་པ་དཀྱིལ་འཁོར་ཀྱི་འཁོར་ལོ་རྫོགས་པར་བསྒོམ་པ་ལ་སྐུ་གསུམ་དང་སྦྱར་བ་དང༌། གསུམ་པ་སྒྲུབ་པ་པོ་བློའི་སྤྱོད་ཡུལ་རྫོགས་པ་གཅིག་ལ་སྐུ་གསུམ་དུ་སྦྱར་བ་དང༌། བཞི་པ་འཁོར་ལོ་གསུམ་གཙོ་བོ་འཕོ་བའི་དབང་དུ་བྱས་ནས་སྐུ་གསུམ་དང་སྦྱར་བ་དང༌། ལྔ་པ་ཨ་བྷི་དྷ་ན་ལྟར་གཙོ་བོའི་སྙིང་པོའི་སྒྲུབ་ཐབས་རང་ལ་ཆོ་ག་རྒྱས་བསྡུས་གསུམ་ལ་སྐུ་གསུམ་དུ་གཞག་པའོ། ། གཉིས་པ་རྫོགས་པའི་རིམ་པ་དང་སྦྱར་བ་ལ་གཉིས་ཏེ། ཕྱག་རྒྱ་བཞིའི་རྫོགས་རིམ་དང་སྦྱར་བ་དང༌། རིམ་པ་བཞི་དང་སྦྱར་བའོ། ། སྤྱི་དོན་གཉིས་པ་བཅོམ་ལྡན་འདས་ཀྱི་སྙིང་པོའི་ཐུན་མོང་གི་དངོས་གྲུབ་ལ་གཉིས་ཏེ།

མནན་པའི་སྒོ་ནས་གསད་པའོ།། །། ལེའུ་བཞི་བཅུ་རྩ་བདུན་པ་ལ་གཉིས་ཏེ། གཞུང་དང་མཚན་ནོ། ། གཞུང་དོན་ལ་གཉིས་ཏེ། བཤད་པར་དམ་བཅའ་བའི་སྒོ་ནས་སྔགས་ཀྱི་ཕན་ཡོན་བསྟན་པ་དང༌། གཞུང་བཤད་པ་དངོས་སོ། ། འདི་ལ་ལས་ཚོགས་བཅུ་གཅིག་སྟེ། ། རིགས་རྒྱུད་བཅད་དང་ཞིང་སོགས་བརླག་པར་བྱ་བ་དང༌། ། དབང་དུ་བྱ་དང་དགུག་གནོན་ལ་སོགས་དང༌། ། སྐྱེས་པ་བུད་མེད་དབང་དུ་བྱ་བ་དང༌། །
ཁྱིར་བྱ་བ་དང་རྫས་མིན་བྱ་བ་དང༌། ། གཟེར་སོགས་ཐོག་འབེབས་བསྐྱེད་པར་བྱ་བ་དང༌། ། བྱི་ལར་བྱ་དང་རྗེས་གཟུང་སྣ་ཚོགས་གཟུགས་སུ་བྱ། ། བཅུ་གཅིག་གཞན་གྱི་འབྲུ་རྣམས་དགུག་པར་བྱ་བའོ།། །། ལེའུ་བཞི་བཅུ་རྩ་བརྒྱད་པ་ལ་གཉིས་ཏེ། གཞུང་དང་མཚན་ནོ། ། གཞུང་ལ་བཞི་སྟེ། ཡུམ་གྱི་སྙིང་པོའི་སྔགས་ཀྱི་ཆེ་བ་བསྟན་པ་དང༌། ཧ་བརྒྱད་ཀྱི་ཕན་ཡོན་བསྟན་པ་དང༌། དངོས་གྲུབ་བསྒོམ་པ་ལ་ལྟོས་དགོས་པས་བསྐྱེད་པའི་རིམ་པ་བསྟན་པ་དང༌།
ལྷ་སུམ་ཅུ་རྩ་བདུན་བསྒོམ་པ་དང༌། མཆོད་པ་འཐད་པ་དང་ཕན་ཡོན་བསྟན་པའོ།། །། ལེའུ་བཞི་བཅུ་རྩ་དགུ་པ་ལ་གཉིས་ཏེ། གཞུང་དང་མཚན་ནོ། ། གཞུང་ལ་དྲུག་སྟེ། མཚམས་སྦྱར་བའི་སྒོ་ནས་སྐྱེ་བདུན་ཤའི་ཕན་ཡོན་དང༌། དེ་བརྟག་པའི་ཐབས་དང་མཚན་ཉིད་དང༌། དེའི་ཤ་བཟའ་བའི་ཕན་ཡོན་དང༌། དེའི་གོ་རོ་ཙ་ན་ལ་སོགས་པ་ལ་བརྟེན་ནས་བསྒྲུབ་པ་དང༌། དེ་ཡང་དཀའ་ལ་རྣམ་སྨིན་ཆེ་བས་གང་རུང་རུང་དུ་མི་བྱ་བ་དང༌།
དེ་བསྒྲུབས་ནས་ཤ་ཟོས་པའི་ཕན་ཡོན་བསྟན་པའོ།། །། ལེའུ་ལྔ་བཅུ་པ་ལ་གཉིས་ཏེ། གཞུང་དང་མཚན་ནོ། ། གཞུང་ལ་གཉིས། མཚམས་སྦྱར་བའི་སྒོ་ནས་སྦྱིན་སྲེག་གི་ཆེ་བའམ་ཕན་ཡོན་བསྟན་པ་དང༌། གཞུང་བཤད་པ་དངོས་སོ། ། དེ་ལ་གཉིས་ཏེ། ལས་ཚོགས་དང་དངོས་གྲུབ་དེ་དག་ཀྱང་བསྐྱེད་པའི་རིམ་པ་བསྒོམ་པ་ལས་བྱུང་བ་བསྟན་པའོ། ། དང་པོ་ལ་བཅུ་གཉིས་ཏེ། སྦྱིན་སྲེག་སྐྱུགས་པ་ཟླ་མཚན་དང༌། །
དབྱེ་བསྐྲད་བར་ཆད་བཟློག་པ་དང༌། ། རིགས་ཀྱི་དབུལ་བ་དོར་བ་དང༌། ། དབང་དུ་བྱ་བ་ལ་སོགས་བསྟན། ། གོ་མཾ་ས་དང་བིལ་བིལ་བ། ། གོ་མཾ་ས་དང་མ་གོ་པཱུ། ། དེས་ཀྱང་དབང་དུ་བྱ་བ་དང༌། ། གཞན་བརྟེན་བརྟུལ་ཞུགས་འདིར་མ་གསུངས། ། བཅུ་གཉིས་གདུག་པ་བཅིང་བ་བསྟན། ། གཉིས་པ་དངོས་གྲུབ་བསྐྱེད་པའི་རིམ་པ་བསྒོམ་པ་ལས་འབྱུང་བར་བསྟན་པ་གཉིས་བསྐྱེད་རིམ་མདོར་བསྟན་ཅིང་ཕན་ཡོན་བསྟན་པ་དང༌། རྒྱས་པར་བཤད་པའི་ཚུལ་གྱིས་གཙོ་བོའི་ལུས་ལ་གནས་པར་བསྟན་པ་དང༌།
དེ་དག་ཀྱང་ལམ་བསྐྱེད་རྫོགས་གཉིས་ཀྱིས་འགྲུབ་པར་གསུངས་པའོ། ། ཐམས་ཅད་ཀྱི་མཇུག་སྡུད་ལེའུ་ལྔ་བཅུ་རྩ་གཅིག་པ་འདི་ལ་གཉིས་ཏེ། གཞུང་དང་མཚན་ནོ། ། གཞུང་ལ་གཉིས་ཏེ། མཚམས་སྦྱར་བ་དང༌། གཞུང་བཤད་པ་དངོས་སོ། ། འདི་ལ་གསུམ་སྟེ། བརྡའི་སྒོ་ནས་ལམ་གྱི་གསང་བའི་མཇུག་བསྡུ་བ་དང༌། ལམ་གོམས་པས་ཕན་ཡོན་གྱི་སྒོ་ནས། འབྲས་བུའི་གསང་བ་མཇུག་བསྡུ་བ་དང༌།
རྒྱུད་འདི་གཞན་ལས་ཁྱད་པར་དུ་འཕགས་པ་བསྟན་པའོ། ། དང་པོ་ལ་གསུམ་སྟེ། དེ་ཁོ་ན་བཅུ་བཞིས་བསྐྱེད་རིམ་གྱི་མཉམ་གཞག་དང་རྗེས་ཐོབ་ཏུ་བཤད་པ་དང༌། སྒོང་སྐྱེས་ལྟར་ན་བསྐྱེད་རྫོགས་གཉིས་ཀ་བསྟན་པ་དང༌། གཉིས་ཀའི་ཐུན་མོང་གི་མཇུག་སྡུད་ངེས་ཚིག་དང་བཅས་པ་བསྟན་པའོ། ། གཉིས་པ་ལམ་གྱི་གོམས་པའི་ཕན་ཡོན་བསྟན་པ་ལ་དྲུག་སྟེ། ཕན་ཡོན་དངོས་སུ་འགྲོ་བའིསྒོ་ནས་བསྟན་པ་དང༌། ཕན་ཡོན་དང་ལྡན་པའི་ལམ་བསྟན་པ་དང༌། ལམ་གྱི་ཁྱད་པར་རྒྱན་དང་ལྷའི་རང་བཞིན་བསྒོམ་བྱར་བསྟན་པ་དང༌། རྒྱན་སོགས་སྣ་ཚོགས་སུ་བསྟན་ཡང་གདུལ་བྱའི་དབང་གིས་དགོས་པ་དང་བཅས་པར་བསྟན་པ་དང༌།
