དེ་བཞིན་དུ་ཆུས་བཀྲུས་པ་ཡང་དང་ཡང་དུ་བྱས་ལ་ཆུ་དེས་ནད་པ་ལ་ཁྲུས་བྱ་བ་དང་ཁར་ཡང་བཏུང་བར་བྱས་ན། ནད་ཞི་བར་བྱེད་དོ། ། དེའི་སྔགས་ནི་འདི་ཡིན་ཏེ། ཨོཾ་རུ་རུ་ཕུ་རུ་ཛྭ་ལ་ཏིཥྛ་སིདྡྷི་ལོ་ཙཱ་ནི་སརྦ་ཨརྠ་ས་དྷ་ནི་མ་མ་ཤཱནྟིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་བྱའོ། ། ནད་ཞི་བར་བྱ་བའི་ཆོ་ག་བརྗོད་ནས་དུས་མ་ཡིན་པར་འཆི་བ་དང་རྨི་ལམ་ངན་པ་བཟློག་པའི་ཕྱིར་ཐུགས་རྗེ་ཆེན་པོ་ཡི་གེ་ཕྱེད་དང་བཞི་པའི་རྣལ་འབྱོར་གྱི་ཐབས་བཤད་པར་བྱའོ། ། དེ་ཡང་འདི་ཟླ་བ་ཡར་ངོ་ལ་གནས་ཡིད་དུ་འོང་བར་བསྙེན་གནས་ཁྲུས་དང་གཙང་སྦྲ་དང་བཅས་ཏེ་གོས་གསར་པ་གྱོན་ལ་ཟས་དཀར་གསུམ་ལ་བརྟེན་ཏེ། ཐ་མལ་པ་དེ་ཉིད་ཀྱིས། གསང་སྔགས་བཟླས་ཏེ་གྲངས་བཟུང་བའམ། ཡང་ན་གྲངས་ངེས་པ་མེད་པར་ཅི་ནུས་པར་བཟླས་པར་བྱའོ། །

ཏདྱ་ཐཱ། ཨོཾ་མེ་རུ་མེ་རུ། མ་ཧཱ་ཡ་མེ་རུ་ཡེ་སྭཱ་ཧཱ། ཞེས་པའོ། ། དེ་ལྟར་ཚེ་རིང་བར་འགྱུར་བའི་ཆོ་ག་བརྗོད་ནས་ད་ནི་ཁྱད་པར་ལས་ཀྱི་སྒྲིབ་པ་ཟད་པར་བྱ་བའི་ཕྱིར་ཡི་གེ་བརྒྱ་པའི་རྣལ་འབྱོར་ལ་གནས་པར་བྱའོ། ། དེ་ལས་འདི་ལྟར་སྔར་བརྗོད་པ་རྣམས་ཀྱིས་ཀྱང་ལས་ཀྱི་སྒྲིབ་པ་འདག་ཅིང་མངོན་པར་བྱང་ཆུབ་པར་འགྱུར་མོད་ཀྱི་གཙོ་བོར་སྡིག་པ་འདག་པར་བྱེད་པ་ཡིན་པས་འདི་ལ་འབད་པར་བྱས་ན་མྱུར་དུ་བྱང་ཆུབ་ཐོབ་པར་འགྱུར་རོ། ། དེའང་འདི་ལྟར་དཀར་པོའི་ཕྱོགས་ཀྱི་ཚེས་གཅིག་གི་ནང་པར་ནས་བརྩམས་ཏེ། གནས་ཡིད་དུ་འོང་ཞིང་གཙང་ལ་དྲི་ཞིམ་པོ་དང་ལྡན་པའི་སར། ཁྲུས་དང་གཙང་སྦྲ་ལ་གནས་པས་ཨ་མོ་གྷ་པཱ་ཤ་ལ་སོགས་པའི་གསོ་སྦྱོང་དང་ལྡན་པས་རང་བཞིན་གྱིས་ཁ་ན་མ་ཐོབ་ལ་སོགས་པ་དང༌། ཤ་དང་ཆང་སྤངས་ཏེ།
དཀར་གསུམ་ལ་གནས་དེས་དྲི་ཞིམ་པོའི་ཆུས་མཎྜལ་ལེགས་པར་བྱུགས་ཏེ། དེའི་སྟེང་དུ་སྟེགས་བུ་ཅུང་ཟད་མཐོ་བའི་སར་ཤཱཀྱ་ཐུབ་པའི་སྐུ་ལུགས་མ་ལ་སོགས་པ་བཞག་ལ། དེའི་མདུན་དུ་རང་ཅུང་ཟད་དམའ་བའི་སྟན་ལ་པུས་མོ་གཉིས་བཙུགས་ཏེ་བསྡད་ལ། བདག་གི་སྡིག་པ་དག་པར་བྱས་ལ། རྫོགས་པའི་སངས་རྒྱས་ཀྱི་གོ་འཕང་ཐོབ་པར་བྱས་ལ་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་བྱའོ། ། ཞེས་སེམས་བསྐྱེད་པར་བྱའོ། ། ཡན་ལག་བདུན་པ་བཟང་སྤྱོད་བཞིན་བྱ་གསུང་ངོ༌། ། དེ་ནས་སྐུ་དེ་ཉིད་དམིགས་ཏེ་གསང་སྔགས་བཟླ་བར་བྱ་བ་ཡིན་ཏེ། དེ་ལ་སྔགས་རེ་རེའི་སྔོན་ལའང་གུས་པ་དང་བཅས་པས་བཏུད་དེ་ཕྱག་བྱའོ། ། དེ་ལྟར་སྔགས་རེ་རེ་ལ་ཕྱག་རེ་རེ་མཆོད་པ་བཟང་པོ་ཉིན་མོ་ལན་གསུམ་དུ་བྱ་ཞིང་བཟླའོ། །
མཚན་མོ་ལན་གསུམ་མ་གྲུབ་ནའང་སྲོད་དང་ཐོ་རངས་གཉིས་སུ་བྱའོ། ། དེ་ལྟར་མཆོད་པ་དང༌། ཕྱག་དང་བཟླས་པ་རྣམས་ཀྱི་ཚུལ་གྱིས་ཉིན་མཚན་མེད་པར་བྱས་ན་མཚན་མ་འབྱུང་བར་ཐེ་ཚོམ་མེད་དོ། ། དེ་ལ་མཚན་མ་ནི་དངོས་སུའམ། རྨི་ལམ་དུ་དགེ་སྦྱོང་གི་གཟུགས་སམ་སྐུ་ཉིད་དམ་གང་ཡང་རུང་བའི་གཟུགས་སུ་འོངས་ནས་རིགས་ཀྱི་བུ་ཁྱོད་ལ་སྡིག་པ་མེད་དོ་ཞེས་སྨྲ་བར་འགྱུར་རོ། ། མཚན་མ་འབྱུང་བའི་དུས་ནི་ཞག་བཅུའམ་ཟླ་བ་གཅིག་གམ་གཉིས་སམ་གསུམ་ལ་སོགས་པ་ན་མཚན་མ་འབྱུང་སྟེ། ཇི་སྲིད་མཚན་མ་མ་བྱུང་གི་བར་དུ་བྱ་སྟེ། མཚན་མ་འབྱུང་བ་ལ་ཐེ་ཚོམ་མི་ཟའོ། ། ཟླ་བ་ལ་སོགས་པ་ལ་བྱེད་ནའང་འདི་ནི་ཡར་ངོ་ལ་བྱའི་མར་ངོ་ལ་མི་བྱའོ། ། ངེས་པའི་རྟགས་འབྱུང་བ་ཡིན་ནོ། །
འདི་ལ་བདེ་བར་གཤེགས་པ་ཤཱཀྱ་ཐུབ་པའི་སྐུ་མེད་དུ་མི་བཏུབ་བོ། ། དེ་ལ་གསང་སྔགས་འདི་ཡིན་ཏེ། ན་མ་སྟྲཻ་ཡཱ་དྷྭི་ཀཱ་ནཱཾ་ཏ་ཐཱ་ག་ཏཱ་ནཱཾ། སརྦ་ཏྲཱ་པྲ་ཏི་ཧ་ཏཱ་པཱཔྟི་དྷརྨ་ཏཱ་བ་ལི་ནཱཾ། ཨ་ས་མ་ས་མ་ས་མནྟ་ཏོ྅་ནནྟྲཱ་པཱཔྟི་ཤཱ་ས་ནི། ཧ་ར་ཧ་ར་སྨ་ར་སྨ་ར་ཎ། བི་ག་ཏ་རཱ་ག་བུདྡྷ་དྷརྨ་ཏེ། ས་ར་ས་ར་ས་མ་བ་ལ། ཧ་ས་ཧ་ས། ཏྲ་ཡ་ཏྲ་ཡ།
