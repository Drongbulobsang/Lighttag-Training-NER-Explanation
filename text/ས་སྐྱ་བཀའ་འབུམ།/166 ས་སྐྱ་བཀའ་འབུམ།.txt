རྡོ་རྗེའི་རྩེར་ལྷན་ཅིག་སྐྱེས་པའི་ཡེ་ཤེས་བསྐྱེད་ལ། བྱང་ཆུབ་ཀྱི་སེམས་ཉམས་སུ་མི་གཞུག་པར་སླར་ལུས་བསྒྲིམ་ཤེས་པ་ཁོང་ཀློད། ཧཱུྃ་ངམ་སིའི་སྒྲས་དྲངས་ལ་ལུས་ཐམས་ཅད་བདེ་སྟོང་གིས་རྒྱས་པར་བྱས་ལ། དེའི་ངང་ལ་བློ་བཞག་ཅིང་བསྒོམ་པར་བྱའོ། ། བདེ་བའི་ཉམས་ཆུང་ན་སྔར་གྱི་བཞིན་ཉམས་སུ་བླང་ཞིང་བསྒོམས་ཏེ། ལས་ཀྱི་ཕྱག་རྒྱ་གོམས་པའི་ཚུལ་དུ་བསྒོམ་པའོ། ། གཉིས་པ་ནི་སྙོམས་འཇུག་སྤངས་ལ་འཁྲིག་པའི་གཏམ་ལ་སོགས་པ་ལ་བརྟེན་ནས་བདེ་བ་ཆེན་པོའི་ཡེ་ཤེས་དང༌། རླུང་བསྒོམ་པ་ལ་སོགས་པ་ཉིད་ཀྱི་བྱང་ཆུབ་ཀྱི་སེམས་ཟིན་པ་དང༌། བདེ་བ་ཆེན་པོའི་ཡེ་ཤེས་གོང་བཞིན་དུ་རྒྱས་པར་འགྱུར་ཏེ། དམ་ཚིག་གི་ཕྱག་རྒྱ་གཟུང་འཛིན་གྱི་ཚུལ་དུ་བསླབ་པའོ། །
གསུམ་པ་ནི། བདག་དོན་ཡོད་གྲུབ་པའི་ང་རྒྱལ་དང་ལྡན་པས། མདུན་དུ་ཡིད་ལས་བྱུང་བའི་མིའི་བུ་མོ་སྒྲོལ་མར་བསྐྱེད་པའི་འོད་ཀྱི་རིག་མ་དང་སྙོམས་པར་འཇུག་ཅིང་བདེ་བ་ཆེན་པོའི་ཡེ་ཤེས་བསྒོམ་པ་སྟེ། ཆོས་ཀྱི་ཕྱག་རྒྱ་ཡིད་ལས་བྱུང་བའི་ཚུལ་དུ་བསྒོམ་པའོ། ། གསུམ་པ་གང་ཡང་རུང་བ་ཅིག་གིས་ཕྱག་རྒྱ་ཆེན་པོ་མཚོན་ནུས་ན་གཉིས་པ་བསྒོམ་མི་དགོས་སོ། ། བཞི་པ་ནི་གོང་མ་རྣམས་ཀྱིས་ཉམས་སུ་མྱོང་པ་དང༌། སྣང་བའི་དངོས་པོ་རོ་གཅིག་པར་བསྲེས་ལ། ཆོས་ཐམས་ཅད་སྤྲོས་པའི་མཁའ་ཐམས་ཅད་དང་བྲལ་བ་འབྲས་བུ་སངས་རྒྱས་ཆོས་ཀྱི་སྐུ་དང༌། རང་གི་སེམས་ཀྱི་ཆོས་ཉིད་གཉིས་སུ་མེད་པའི་ཚུལ་བདེ་སྟོང་དབྱེར་མེད་ཀྱི་ངོ་བོ་ཡིད་ལ་བྱས་ཤིང་བསྒོམ་པ་བྱས་ན། རིང་པོར་མི་ཐོགས་པར་ཆོས་ཀྱི་སྐུ་ཐོབ་པར་འགྱུར་ཏེ།

བརྒྱད་ཡོད་ན་བང་མཛོད་ལ་སོགས་པའི་ཕྱོགས་བརྒྱད་དུ་བཞག གཅིག་ལས་མེད་ན་ཁང་པའི་དབུས་སུ་ཅུང་ཟད་མཐོ་བའི་སྟེགས་བུའི་དབུས་སུ་ལེགས་པར་སྦས་ལ། ཟླ་བ་ཡར་གྱི་ངོ་རྣམས་ལ་མཆོད་པ་རྒྱ་ཆེན་པོ་བྱས་ན་ལོངས་སྤྱོད་རྒྱ་ཆེན་པོ་འཕེལ་བར་འགྱུར་རོ། ། འདི་ནི་སྔོན་དུ་བསྙེན་པ་རྫོགས་པའི་ལས་ཡིན་ནོ། ། འཕགས་མ་སྒྲོལ་མ་འགུགས་ཤིང་ཕོངས་པ་ཐམས་ཅད་སེལ་བའི་ལས་ཀྱི་ཡན་ལག་ལོངས་སྤྱོད་འཕེལ་བ་སྟེ། བཅུ་གཅིག་པའོ།། །། བཅུ་གཉིས་པ་སྦྱིན་སྲེག་གི་བསྙེན་པ། སྔོན་དུ་སོང་ནས་བྱས་ན་དངོས་གྲུབ་ཐམས་ཅད་འབྱུང་བའི་རྩ་བ་ཡིན་ནོ། ། དེ་ཡང་རྒྱལ་ལམ། སྟའི་ཟླ་བ་ལ་སོགས་པ་ཤིས་པའི་ཡར་ངོའི་ཚེས་གཅིག་ནས་ཉའི་བར་དུ་བྱ་བ་ཡིན་ཏེ།

དེ་རྫ་ཕོར་སོ་མ་བཏང་བའི་ནང་དུ་བཅུག་ནས་རས་དཀར་པོས་ཁ་བཅད་ལ་ཚོན་སྐུད་སྣ་ལྔའི་གཟུངས་ཐག་བཏགས་ལ་མཎྜལ་ཁར་བཞག་ལ་རྣམ་པར་ཤེས་པ་དགུག་གཞུག་བྱས་ཏེ་བསྟིམ་པ་ལས་བདུག་པས་བདུག ཤིང་ཚེར་མ་ཅན་ལ་མེ་སྦར་ལ་ལིངྒ་གཏུབས་ལ་མིང་དང་མཱ་ར་ཡའི་སྤེལ་ཚིག་དང་བཅས་པས་བསྲེགས་ན་གནོད་བྱེད་འཆི་བར་འགྱུར་རོ། ། འདི་ནི་གནོད་སྦྱིན་ལྕགས་ཀྱུའི་གདུག་པ་ཞེས་བྱ་བ་ཡིན་ནོ་ཞེས་གསུངས་སོ། ། འཕགས་མ་སྒྲོལ་མ་ཡོངས་སུ་སྨིན་པར་བྱེད་པ་ཞེས་བྱ་བའི་ལས་ཀྱི་ཡན་ལག་འབྱུང་པོ་བཟློག་པ་སྟེ། བཅུ་གསུམ་པའོ།། །། བཅུ་བཞི་པ་བདག་གཞན་སྲུང་བའི་འཁོར་ལོ་ནི། བདག་ལྷའི་ང་རྒྱལ་དང་ལྡན་པས་བྲི་བའི་གཞི་དཀར་ཞིང་འཇམ་པའི་དར་རམ་རས་ལ་སྣག་ཚ་ནི་སྐྱེས་པ་བསྲུང་བ་ལ་གུར་གུམ། བུད་མེད་ལ་གི་ཝང་གིས་འཁོར་ལོ་རྩིབས་བརྒྱད་པ་ལྟེ་བ་དང་བཅས་པ་མུ་ཁྱུད་རིམ་པ་བདུན་གྱིས་བསྐོར་བའི་དབུས་ཀྱི་ལྟེ་བ་ལ་ཨོཾ་ཆེ་གེ་མོ་རཀྵ་རཀྵ་ཧཱ་བྲི། རྩིབས་བརྒྱད་ལ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སྭཱ་བྲི། མུ་ཁྱུད་ནང་མ་ལ་རྟེན་འབྲེལ་སྙིང་པོ་བྲི།

ནད་པ་ལ་བྱ་བ་དང༌། གཞན་ལ་བྱ་བའོ། ། དང་པོ་ནི་ཟངས་གཞོང་ལ་སོགས་པའི་སྟེང་དུ་བུམ་པ་བླངས་ལ། ནད་པ་སྟེགས་བུ་གཅིག་གི་སྟེང་དུ་གོས་ངན་པ་བསྐོན་ལ་བཞག་ལ་བུམ་པའི་ཆུས་བཀྲུ། རྨ་བྱའི་མདོངས་ལ་སྔགས་བཏབ་པས་བྱབས། སྔགས་དང་བཀྲ་ཤིས་ཀྱི་ཚིགས་སུ་བཅད་པ་བརྗོད། སེམས་ཀྱིས་བ་སྤུའི་བུ་ག་ཐམས་ཅད་ནས་ནད་གདོན་ཐམས་ཅད་ནག་ཉི་ལི་ལི་འགྲོ་བར་བསམས་ལ། བཀྲུས་ནས་གོས་ངན་པ་དེ་དོར་ཏེ་གོས་གསར་པ་བསྐོན་ནས་བཞག་ན་སྡུག་བསྔལ་ཞི་བར་འགྱུར་རོ། ། ནད་པ་མ་ཡིན་པ་ལ་བྱབས་པ་ནི། ལྷར་བསྐྱེད་པ་དང་གོས་ངན་མི་བསྐོན་པ་ཡིན་མོད་ཀྱི་ཆོ་ག་ལྷག་མ་རྣམས་ནི་སྔ་མ་བཞིན་ནོ། །

བཟླས་བསྒོམ་སྔ་མ་ལྟར་བྱ་ཞིང་ཐོ་རངས་ཀྱི་ཐུན་གྱི་ཁྱེ་སུས་ལ་ལན་བདུན་བདུན་བཟླས་ཤིང་ཅི་རིགས་པ་ཟོས་ན་ཟླ་བ་གཅིག་གིས་རྩོད་པའི་ལན་ཐེབས་པར་འགྱུར་བ་དང༌། གསུམ་གྱིས་ཚིགས་སུ་བཅད་པ་རྩོམ་པ་དང༌། དྲུག་གིས་སྒྲོལ་མ་ཉིད་མངོན་སུམ་དུ་མཐོང་བར་འགྱུར་རོ། ། ཡང་གཅིག་གིས་ཐོས་པ་འཛིན་པར་འགྱུར་བ་དང༌། གསུམ་གྱིས་དབང་ཕྱུག་ཆེན་པོ་ལྟར་སྒྲ་རྩོམ་པར་འགྱུར་བ་དང༌། དྲུག་གིས་ཚངས་པའི་དབྱངས་ཐོབ་པར་འགྱུར་རོ། ། དུས་ངེས་པ་མེད་པར་རྒྱུན་དུ་བསྒྲུབ་པ་ནི། གངས་ལས་སྐྱེས་པའི་སེར་པོ་ཁྲག་རྐང་དང༌། ཤུ་དག་ཆ་མཉམ་པར་ཞིབ་པར་བཏགས་པ་ལ་ལྷའི་རྣལ་འབྱོར་སྔར་བཞིན་དུ་བྱས་ནས་ཐོ་རངས་ཀྱི་ཆ་ལ་ལན་བདུན་བཏབ་ཅིང་ཟོས་ན་ཟླ་བ་གཅིག་གིས་བདག་ཡིད་གཞུངས་ཤིང་དབྱངས་སྙན་པར་འགྱུར་བ་དང༌། གསུམ་གྱིས་ཤེས་རབ་དང་ལྡན་པ་དང༌།

བསྟོད་པ་རྩ་གཅིག་པས་བསྟོད་མཆོད་པ་བྱས་ལ་བདག་འཁོར་དང་བཅས་པའི་ལམ་གྱི་འཇིགས་པ་ཆོམ་རྐུན་ལ་སོགས་པ་འཇིགས་པ་བཅིང་བ་དང༌། སྲུང་བར་མཛད་དུ་གསོལ་ཞེས་གསོལ་བ་བཏབ་ནས། སྔགས་བཟླ་ཞིང་འགྲོ་བར་བྱ་སྟེ། དེའི་དུས་སུ་ཡང་རྗེ་བཙུན་མ་བདག་ལ་ཞལ་གཟིགས་པ་སྐྱབས་སྦྱིན་གྱི་ཕྱག་རྒྱས་བདག་གི་སྤྱི་བོར་རེག་ཅིང་མ་འཇིགས་ཤིག་གསུང་བར་བསམ། འདི་ཆོམ་རྐུན་བཅིང་བ་གཅིག་པུར་མ་ཟད་དེ། བཙོན་ར་ལས་ཐར་བ་དང༌། གཅན་གཟན་ལ་སོགས་པ་གང་ཡང་བསྲུང་བར་ནུས་སོ། ། འཕགས་མ་སྒྲོལ་མ་བདེ་བ་ཐམས་ཅད་སྒྲུབ་པའི་ལས་ཀྱི་ཡན་ལག་ཆོམ་རྐུན་བཅིང་བ་སྟེ། བཅུ་བདུན་པའོ།། །། བཅོ་བརྒྱད་པ་ཀླུ་ལ་སོགས་པའི་གནོད་པ་བསྲུང་བ་ནི་བསྙེན་པ་སྔོན་དུ་སོང་བས་བྱ་བ་ཡིན་ཏེ།
