ཆོ་ག་རྒྱས་པ་ཐབས་ཆེ། བསྡུས་པ་ཆུང་ན་རྒྱས་པ་ཁོ་ན་ཉམས་སུ་བླངས་ལ་འབྲས་བུ་མྱུར་དུ་འབྱུང་སྟེ། ཐབས་ཆེ་ཆུང་སོགས་མེད་དོ། ། འོན་ཀྱང་རྒྱས་བསྡུས་སུ་འོང་བའི་གནད་ཀ་ནི། གདུལ་བྱ་རྒྱས་པ་ལ་དགའ་བ་དང་བསྡུས་ལ་དགའ་བའི་དོན་དུ་བཅོམ་ལྡན་འདས་ཀྱིས་རྒྱས་བསྡུས་སུ་གསུངས་ལ། དེའི་རྗེས་སུ་ཞུགས་པའི་པཎྜི་ཏ་རྣམས་ཀྱིས་ཀྱང༌། རྒྱས་བསྡུས་སུ་གསུངས་པའོ། ། འོ་ན་གང་ཟག་བསྡུས་པ་ལ་དགའ་བའི་དོན་དུ་བཅོམ་ལྡན་འདུས་ཀྱིས་བསྡུས་པ་གསུངས་པ་དེ་རྗེས་ཐོབ་ཀྱི་ཤེས་པ་མངའ་འམ་མི་མངའ་བྱས་པ་ལ། བླ་མ་ཁམས་པ་རྗེས་ཐོབ་ཀྱི་ཤེས་པ་མི་བཞེད་དོ། ། འོ་ན་བསྡུས་པ་ལ་དགའ་བའི་གང་ཟག་ལ་འདི་བསྡུས་པ་ལ་དགའ་བར་ཤེས་པ་དེ་རང་རྗེས་ཐོབ་ཀྱི་ཤེས་པ་མིན་ནམ་ཞེ་ན།

དེ་བཞིན་དུ་ས་བཅད་ཀུན་ལ་ཤེས་པར་བྱའོ། ། རྡོ་རྗེ་གདན་ལྟ་བུ་ཅིག་ལ། མངོན་རྟོགས་ཤེས་ན་ཀུན་ཤེས་པས། རྡོ་རྗེ་གདན་ལྟ་བུ་དེ་བསྒྲུབ་པར་འདོད་པས་དབང་བསྐུར་ཐོབ་པ་རྡོ་རྗེ་གདན་པའི་ཆོས་ལྔ་ཤེས་པར་བྱས་ནས་ཉམས་དགའ་བར་གནས་ནས་དང་པོ་ཁྲུས་བྱས་ལ། གནས་ཁང་བྱི་དོར་བྱས་ཏེ་བདག་གིས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དུ་རྫོགས་པའི་སངས་རྒྱས་ཐོབ་པར་བྱའོ་སྙམ་པའི་སེམས་བསྐྱེད་ལ་དངོས་སུ་བསག་པར་བྱ་བ་ནི། མཎྜལ་བྱུག་པར་བྱ་བའི་ས་ཕྱོགས་སུ་ལག་པ་གཡས་པ་ས་གནོན་གྱི་ཚུལ་དུ་བཞག་ལ། ཨོཾ་བཛྲ་བྷཱུ་མི་ཧཱུྃ་ཞེས་པས་ས་དེ་གཟུང་བར་བྱའོ། ། དེ་ནས་ཨོཾ་བཛྲ་རེ་ཁེ་ཧཱུྃ། ཞེས་པས་ས་དེ་བྱུག་པར་བྱའོ། ། དེ་ནས་ལག་པ་འདེགས་དུས་སུ་ལག་པ་གཡས་ཀྱི་སྲིན་ལག་གི་སྟེང་དུ་གཡོན་པས་མེ་ཏོག་དེ་བཞག་ནས་གཡས་པའི་སྟེང་ནས་མས་རིམ་གྱིས་སོང་སྟེ་མཎྜལ་གྱི་སྟེང་དུ་བོར་བ་དེས་བསྲུང་ངོ༌། །

དངོས་སུ་ལུས་ངག་གི་མཆོད་རྫས་སོགས་ཕུལ་གྱིན་བྱས་པ་བས་བསོད་ནམས་ཆེ་སྟེ། དངོས་སུ་མཆོད་རྫས་ཤིན་ཏུ་མང་བར་བཤམས་ནས་ཕུལ་ཡང་དེ་ལ་ཅང་མེད་དེ། ཡིད་ཀྱིས་མེ་ཏོག་སོགས་འཇིག་རྟེན་གྱི་ཁམས་གང་བ་ནས་མར་མེས་འཇིག་རྟེན་གྱི་ཁམས་ཀྱི་མུན་པ་ཐམས་ཅད་བསལ་བར་བསམ་ཞིང་བྱས་པས་བསོད་ནམས་ཆེ་སྟེ་ཡིད་ཀྱིས་འབྱོར་པ་ཙམ་དངོས་སུ་མི་འབྱོར་རོ། ། ཡང་དངོས་སུ་འབྱོར་པ་བས་ཡིད་ཀྱིས་བསོད་ནམས་ཆེ་སྟེ་བག་ཆགས་ཐམས་ཅད་སེམས་ལ་འཇོག་པས་བསོད་ནམས་ཆེ་སྟེ། དངོས་སུ་བྱས་པ་ལུས་ངག་ལ་བག་ཆགས་བཞག་པ་དེ་སྲོག་གི་དབང་པོ་འགགས་ཙམ་ན་ལུས་ངག་དང་འབྲལ་ཡང་སེམས་དང་མ་བྲལ་བས་སོ་གསུངས་ཏེ། འོ་ན་ཡིད་ཁོ་ནས་མི་བྱེད་པར་དངོས་ཀྱིས་བཤད་འདུག་པ་བྱས་པ་ལ། དེ་ནི་འཇིག་རྟེན་ན་དངོས་པོའི་རྫས་ལ་མངོན་ཞེན་ཆེ་བས་དངོས་པོའི་རྫས་བག་རེ་བཏང་བ་ལ་མཆོག་ཏུ་བྱེད་པས་དེ་དང་བསྟུན་པའོ། ། དེ་ནས་ཡེ་ཤེས་ཀྱི་ཚོགས་བསག་གོ ། བསོད་ནམས་ཀྱི་ཚོགས་ལས་འབྲས་བུ་གཟུགས་སྐུ་གཉིས་འབྱུང་ལ། ཡེ་ཤེས་ཀྱི་ཚོགས་ལས་ཆོས་སྐུ་འབྱུང་བས་སོ། །

བཅུ་པ་རྒྱལ་པོ་རོལ་པ་ནི་ཟླ་བ་ལ་མྃ་ལས་སོ། ། བཅུ་གཅིག་པ་རིག་པ་འཛིན་པའི་སྡེ་སྣོད་ཀྱི་འཇམ་དཔལ་ནི་ཟླ་བ་ལ་ཐལྷིཾ་ལས་སོ། ། བཅུ་གཉིས་པ་རིག་པ་འཛིན་པ་ནི་ཟླ་བ་ལ་དྷཱིཿལས་སོ། ། བཅུ་གསུམ་པ་ཡང་འཕགས་པ་འཇམ་དཔལ་ནི་ཟླ་བ་དང་ཐལྷིཾ་ལས་སོ། ། བཅུ་བཞི་པ་ཡན་ལག་མེད་པའི་འཇམ་པའི་རྡོ་རྗེ་ཟླ་བ་ལ་ཧྃ་ལས་སོ། ། བཅོ་ལྔ་པ་ཆོས་དབྱིངས་ངག་དབང་ནི་ཟླ་བ་དང་འོད་ཟེར་རོ། ། བཅུ་དྲུག་པ་མཚན་བརྗོད་ཀྱི་ནི་ཟླ་བ་དང་མུཿདམར་པོ་ལས་སོ། ། བཅུ་བདུན་པ་འཇམ་པའི་རྡོ་རྗེ་དཔའ་བོ་གཅིག་པ་ནི་ཟླ་དང་མྃ་ལས་སོ། ། བཅོ་བརྒྱད་པ་འཇམ་དཔལ་ཤེས་རབ་འཁོར་ལོ་ནི་ཟླ་བ་དང་ཨ་འམ་དྷཱིཿའོ། ། བཅུ་དགུ་པ་འཕྲལ་དུ་ཉམས་སུ་མྱོང་བ་ནི་ཟླ་བ་དང་དྷཱིཿའོ། །

ཞེ་དགུ་པ་ཤེས་རབ་ཕ་རོལ་ཕྱིན་མ་ནི་པད་ཟླ་ལ་པྲྃ་ལས་སོ། ། ལྔ་བཅུ་པ་ཤེས་རབ་ཕ་རོལ་ཕྱིན་མ་ནི་པད་ཟླ་ལ་བྷཱིཿལས་སོ། ། ང་གཅིག་པ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་ནི་པད་ཟླ་ལ་དྷཱིཿལས་སོ། ། ང་གཉིས་པ་བསྡུས་པའི་ཤེར་ཕྱིན་ནི་པད་ཟླ་ལ་དྷཱིཿསེར་པོ་ལས་སོ། ། ང་གསུམ་པ་མི་གཡོ་བ་ནི་པད་ཉི་ལ་ཧཱུྃ་ལས་སོ། ། ང་བཞི་པ་གཏུམ་མོ་ཆེན་པོ་ནི་པད་དཀར་ཉི་མ་ལ་ཧཱུྃ་ལས་སོ། ། ང་ལྔ་པ་རྡོ་རྗེ་ཧཱུྃ་མཛད་ནི་པད་ཉི་ལ་ཧཱུྃ་ལས་སོ། ། ང་དྲུག་པ་རི་ཁྲོད་ལོ་མ་ཅན་པད་ཟླ་ལ་པྃ་སེར་པོ་ལས་སོ། ། ང་བདུན་པ་དབྱངས་ཅན་མ་ནི་པད་ཟླ་ལ་དོཿལས་སོ། ། ང་བརྒྱད་པ་བགེགས་དགྲ་ནི་པད་ཉི་ལ་ཧཱུྃ་ལས་སོ། །
ང་དགུ་པ་རལ་པ་གཅིག་མ་ནི་པད་ཉི་ལ་ཏཱྃ་ལས་སོ། ། དྲུག་ཅུ་པ་དུག་སེལ་མ་ནི་པད་ཉི་ལ་ཛྃ་ལས་སོ། ། རེ་གཅིག་པ་རིག་བྱེད་མ་ནི་པད་ཉི་ལ་ཧྲཱིཿལས་སོ། ། རེ་གཉིས་པ་ཡང་རྟོག་པ་ལས་གསུངས་པའི་འོད་ཟེར་ཅན་ནི་པད་ཟླ་ལ་ཧྲཱིཿལས་སོ། ། རེ་གསུམ་པ་ཨཽ་ཌྱན་གྱི་རིག་བྱེད་མ་ནི་པད་ཟླ་ལ་ཧྲཱིཿལས་སོ། ། རེ་བཞི་པ་ཨཽ་ཌྱན་ནས་བྱུང་བའི་རིག་བྱེད་མ་ནི་པད་ཉི་ལ་ཧཱུྃ་ལས་སོ། ། རེ་ལྔ་པ་ཁམས་གསུམ་དབང་དུ་བྱེད་པ་ནི་པད་ཉི་ལ་ཧྲཱིཿལས་སོ། ། རེ་དྲུག་པ་གནོད་གནས་ནི་པད་ཟླ་ལ་བྃ་ལས་སོ། ། རེ་བདུན་པ་རྒྱ་ནག་རིམ་པའི་སྒྲོལ་མ་ནི་པད་ཟླ་ལ་ཏཱྃ་ལས་ཐོད་པའི་ནང་དུ་གྲི་གུག་ཧཱུྃ་གིས་མཚན་པ་ལས་སོ། ། རེ་བརྒྱད་པ་རྡོ་རྗེ་སྒྲོལ་མ་ནི་པད་ཟླ་ལ་ཏཱྃ་ལས་སོ། །
