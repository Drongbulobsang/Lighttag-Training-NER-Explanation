གཅིག་ནས་གཉིས་སུ་འཛག་སྟེ་རྒྱལ་པོས་ཐོས་པ་ལས། རྒྱལ་པོ་ན་རེ་བློན་པོ་འདི་སྙིང་དང་འདྲ་བ་ཡིན་པས་འདི་མི་བདེན་མི་སྲིད་བྱས་པ་ལ། གཏམ་མང་དུ་གྲགས་པ་ལ་ད་ནི་བདེན་ནམ་སྙམ་ནས་འདུག་པ་ལ་བློན་པོ་གྲོགས་པོ་ཞིག་གིས་ཐད་ཀ་འདི་སྤོངས། ཁྱོད་ལ་རྒྱལ་པོ་འཁོར་དང་བཅས་པས་ཆད་པ་ཆེན་པོ་གཅོད་པར་འདུག་གོ་བྱས་པས། དེ་བཞིན་དུ་བྱའོ་ཟེར་ནས་རས་མང་པོས་དྲིལ་ནས་རྒྱས་བཏབ་ཏེ་ཉི་མ་ལྷོ་ཕྱོགས་ན་ཕར་བསྐུར་ཏེ་བཏང་བ་ལ། རྒྱལ་པོའི་གྲོང་ཁྱེར་ཕྱོགས་བཞི་ན་སྒོ་བཞི་ཡོད་པའི་སྒོ་བ་རྣམས་ཀྱིས། འདི་ཙམ་གྱི་རྒྱས་འདེབས་པ་འདི་ན་ནོར་མང་པོ་ཡོད་ཟེར་ནས་གཡེངས་པས་བྲིས་སྐུ་དེ་བྱུང་བ་ལ། རྒྱལ་མཚན་གྱི་རྩེ་མོ་ལ་བཏགས་ནས་རྒྱལ་པོའི་གན་དུ་ཁྱེར་བ་ལ། རྒྱལ་པོའི་འཁོར་ཐམས་ཅད་འདི་སྲོག་དང་ཕྲལ་ལ། འདིའི་ནོར་ཐམས་ཅད་དང་དབྲལ་ལོ་ཟེར་བ་ལ།
རྒྱལ་པོ་ན་རེ་ལར་བྱ་བ་ཐམས་ཅད་བརྟགས་ནས་བྱ་དགོས་པས་ད་དུང་ཆད་པས་མི་བཅད་ཀྱིས། འུ་ཅག་ཀྱང་བདེ་མཆོག་གདན་དུ་བྱས་པའི་འཇིགས་བྱེད་དང་དུས་མཚན་སྟེང་དུ་བཏོན་པ་བྲིས་ལ་བདེན་པའི་བྱིན་རླབས་འགྲན་ཟེར་ནས། མུ་སྟེགས་ཀྱི་དེའུ་ལའི་ནང་དུ་བཅུག་སྟེ། ཁྱེད་ཀྱི་ལྷ་ཡང་མཆོད་དོ་ཟེར་ནས་ཞག་བདུན་ན་ཆོ་འབྱེད་པ་ལ། བློན་པོ་ཀུ་ས་ལི་ནཱ་ཐས་སློབ་དཔོན་ལ་ཞག་བདུན་ནས་ཆོ་འབྱེད་པས་ནུས་པ་བསྟན་དགོས་ཞུས་པ་ལ། ནུས་པ་བསྟན་ཏོ་གསུང༌། དེ་ནས་ནམ་ཕྱེད་ན་ཏིར་ཏི་ཀའི་མཚན་མ་འཛིན་པ་ཐོད་པ་འཛིན་པ་གཅིག་བྱུང་བས། ངས་ཚོགས་དཔོན་བྱའམ་ཟེར་བ་ལ། ཁྱོད་ཀྱིས་ཚོགས་དཔོན་མི་བྱེད་དོ་བྱས་པ་ལ། ནམ་མཁའ་ལ་འཕུར་ནས་ཧེ་རུ་ཀར་བྱས་ནས་སོང་བ་ལ།

དེ་ནས་སློབ་དཔོན་གྱིས་གཞུང་དྲུག་བརྩམས་སོ། ། དེ་ནས་མཛེ་མོ་དེ་ཡང་མཛེ་མོ་མིན་ཏེ་རྡོ་རྗེ་ཕག་མོའི་རྣམ་འཕྲུལ་ལོ། ། ཞིང་རྨོད་དེ་ཡང་ཞིང་རྨོད་མིན་ཏེ་ཧེ་རུ་ཀ་ཡི་རྣམ་འཕྲུལ་ལོ། ། མཛེས་ལྡན་བུ་མོ་དེ་ཡང་བུ་མོ་མིན་ཏེ་རྡོ་རྗེ་ཕག་མོའི་རྣམ་འཕྲུལ་ཡིན། ཐོད་པ་ཅན་ཡང་ཐོད་པ་ཅན་མ་ཡིན་ཏེ་ཧེ་རུ་ཀ་ཡི་རྣམ་འཕྲུལ་ཡིན། ཁྱོད་ཀྱི་ཞབས་ཀྱི་ལྗིད་ཀྱིས་མནན་ན་འོག་གི་རླུང་ཡང་འཇིག་འགྱུར་ཞེས་ཏེ་ཞབས་ཀྱི་རྩེ་མོས་མནན། ཕྱག་རྣམས་དྲང་པོར་བརྐྱང་ན་ཕྱོགས་སྐྱོང་རྣམས་ཀྱང་སྐྲག་འགྱུར་ཞེས་ཏེ་ཕྱག་རྣམས་བསྐུམ་པས་གནས། དབུ་ཡི་གཙུག་ཏོར་དྲང་པོར་བསྲངས་ན་ཚངས་པའི་འཇིག་རྟེན་འཇིག་འགྱུར་ཞེས་ཏེ་དབུ་ནི་ཡོ་བར་གནས། རྗེ་བཙུན་ཁྱོད་ཀྱི་གར་ནི་དགོས་པ་དང་ལྡན་མཐོང་བས་བསྟོད། ཅེས་གསུངས་ནས་ཛཱ་ལནྡྷ་རར་དགོན་པ་བྱེད་གསུང་པ་ལ།
ནད་ཀྱིས་ཐེབས་ནས་གཡོག་པོ་རྣམས་གནས་ཉི་ཤུ་རྩ་བཞིར་བཏང་སྟེ། བླ་མའི་བཀའ་བཅག་པ་ལ་བཤགས་པ་བྱས། དེ་ནས་དེ་མེ་ཏོག་དང་སྨན་བླངས་ནས་འོངས་པ་ལ། ལམ་ན་ཁྲུས་ཀྱི་རྫིང་ཡོད་པ་ལ་ཁྲུས་བྱ་ལོང་མེད་དེ་ལ་མོར་འགྲོ་བ་ལ། བུད་མེད་མང་པོ་དེ་ན་ཁྲུས་བྱེད་ཅིང་ཁྱེད་ཅག་ཀྱང་ཁྲུས་གྱིས་ཟེར་བ་ལ་ལོང་མེད་སློབ་དཔོན་ན་བས་མྱུར་དུ་འགྲོ་བྱས་པ་ལ། སློབ་དཔོན་སོས་ཟེར་ནས་ཁྲུས་བྱེད་ཅིང་བསྡད་པ་ལ། སྨན་རྣམས་བུད་མེད་དེ་རྣམས་ཀྱིས་ཁྱེར་ནས་བུད་མེད་རྣམས་གར་སོང་ཆ་མེད། དེར་ཁྱིམ་དུ་ཕྱིན་ཙམ་ན་སློབ་དཔོན་གྲོངས། ང་ལ་མཁའ་འགྲོ་མའི་ཆད་པ་བྱུང་བས་ཚེ་འདི་ལ་མ་ཧཱ་མུ་དྲཱ་མེད་ཀྱང༌། བར་དོར་རྟེན་འདི་ལ་ཞུགས་ནས་ཕྱག་རྒྱ་ཆེན་པོ་ཐོབ་པར་བྱེད་ཀྱིས་ཞག་བདུན་དུ་མ་སྲེགས་ཤིག་གསུང་བ་ལ།

རྡུལ་ཚོན་སྦྱང་ཞིང་བྱིན་གྱིས་བརླབ་པ་ནི། ཚོགས་བསགས་པ་ནས་དབུ་རྒྱན་གྱི་ལྷ་དྲན་པའི་བར་དུ་བསྒོམས་ལ། སྔ་མ་བཞིན་དུ་མཆོད་ཅིང་བསྟོད་དེ་བསྡུའོ། ། དེ་བཞིན་དུ་མདུན་གྱི་ཡང་དབུ་རྒྱན་གྱི་བར་དུ་དྲན་པར་བསྒོམས་ལ། ལྷའི་གྲངས་ཀྱི་གཏོར་མའམ། ཡང་ན་གཏོར་མ་གཅིག་གིས་སརྦ་བིགྷཱན་ན་མ་སྶརྦ་ཏ་ཐཱ་ག་ཏེ་བྷྱོ་བི་ཤྭ་མུ་ཁེ་བྷྱཿ སརྦ་ཐཱ་ཁཾ་ཨུངྒ་ཏེ་སྥ་ར་ཎ་ཨི་མཾ་ག་ག་ན་ཁཾ། གྲིཧྞེ་དཾ་བ་ལིཾ་ཏེ། ཞེས་པས་ལན་གསུམ་དུ་བསྐོར་ཞིང་གཏོར་མ་གཏང་ངོ༌། ། ཏིང་ལོ་དང་ཆང་བུ་ཡང་མ་འབྱོར་ན་ཟུང་གཅིག་བཏང་སྟེ་དཀྱིལ་འཁོར་སོ་སོའི་ལས་ཐམས་ཅད་པའི་སྔགས་ཀྱིས་བསྐོར་ལ་གཏང་ངོ༌། །

དེ་ནས་ཡང་བདུད་རྩིས་བསངས་ཤིང་བསྲུང་བར་བྱའོ། ། དེ་ནས་ལྷར་མོས་ཤིང་གོ་ཆས་བརྒྱན་ལ། སློབ་མའི་གཙོ་བོའི་ལག་པ་བཟུང་སྟེ། ཨོཾ་ཤྲཱི་ཧེ་རུ་ཀ་པི་ཏ་ཅེས་བྱ་བ་ལ་སོགས་པའི་སྔགས་བརྗོད་ལ། དཔའ་བོ་བསྐུལ་ལོ། ། དེ་ནས། གང་གི་དྲིན་གྱིས་བདེ་ཆེན་ཉིད། ། ཅེས་བྱ་བ་བཅུག་ལ། རྣལ་འབྱོར་གནས་ལས་ཞེས་པའི་བར་གྱིས་ཕྱག་འཚལ་ཞིང་སྐྱབས་སུ་འགྲོ་བའོ། ། གསོལ་བ་གདབ།
